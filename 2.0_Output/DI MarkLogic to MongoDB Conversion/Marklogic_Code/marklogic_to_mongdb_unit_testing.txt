import pytest
from pymongo import MongoClient
from pymongo.errors import WriteError

# MongoDB connection setup
client = MongoClient('mongodb://localhost:27017/')
db = client['testdb']

# Sample data for testing
sample_documents = [
    {"_id": "/documents/doc1.xml", "content": "Content of doc1"},
    {"_id": "/documents/doc2.xml", "content": "Content of doc2"},
    {"_id": "/documents
ewdoc.xml", "content": "This is a new document"},
    {"_id": "/documents/doc1.xml", "content": "Updated content here"}
]

sample_example = {
    "_id": "/examples/sample.xml",
    "title": "MarkLogic Querying",
    "body": "Learning XQuery in MarkLogic."
}

@pytest.fixture(scope="module")
def setup_database():
    # Setup the database with initial data
    db.documents.delete_many({})
    db.examples.delete_many({})
    db.documents.insert_many(sample_documents)
    db.examples.insert_one(sample_example)
    yield
    # Teardown the database
    db.documents.delete_many({})
    db.examples.delete_many({})

def test_load_documents(setup_database):
    """Test loading of documents from MarkLogic to MongoDB"""
    documents = list(db.documents.find({}))
    assert len(documents) == 4

def test_insert_document(setup_database):
    """Test insertion of a new document"""
    new_doc = {"_id": "/documents
ewdoc2.xml", "content": "This is another new document"}
    db.documents.insert_one(new_doc)
    inserted_doc = db.documents.find_one({"_id": "/documents
ewdoc2.xml"})
    assert inserted_doc is not None
    assert inserted_doc["content"] == "This is another new document"

def test_update_document(setup_database):
    """Test updating an existing document"""
    db.documents.update_one({"_id": "/documents/doc1.xml"}, {"$set": {"content": "Updated content again"}})
    updated_doc = db.documents.find_one({"_id": "/documents/doc1.xml"})
    assert updated_doc is not None
    assert updated_doc["content"] == "Updated content again"

def test_search_documents(setup_database):
    """Test searching documents using text index"""
    db.documents.create_index({"content": "text"})
    search_results = list(db.documents.find({"$text": {"$search": "doc1"}}))
    assert any(doc["_id"] == "/documents/doc1.xml" for doc in search_results)

def test_delete_document(setup_database):
    """Test deleting a document"""
    db.documents.delete_one({"_id": "/documents/doc2.xml"})
    deleted_doc = db.documents.find_one({"_id": "/documents/doc2.xml"})
    assert deleted_doc is None

def test_query_example(setup_database):
    """Test querying example collection"""
    example_doc = db.examples.find_one({"_id": "/examples/sample.xml"})
    assert example_doc is not None
    assert example_doc["title"] == "MarkLogic Querying"
    assert example_doc["body"] == "Learning XQuery in MarkLogic."

def test_insert_empty_document(setup_database):
    """Edge case: Insert empty document"""
    empty_doc = {"_id": "/documents/empty.xml"}
    try:
        db.documents.insert_one(empty_doc)
        inserted_doc = db.documents.find_one({"_id": "/documents/empty.xml"})
        # Should fail schema validation if validator is enabled, otherwise allow
        assert inserted_doc is not None
    except WriteError:
        assert True  # Expected failure

def test_insert_missing_fields(setup_database):
    """Edge case: Insert document with missing fields"""
    incomplete_doc = {"_id": "/documents/incomplete.xml"}
    try:
        db.documents.insert_one(incomplete_doc)
        inserted_doc = db.documents.find_one({"_id": "/documents/incomplete.xml"})
        assert inserted_doc is not None
    except WriteError:
        assert True  # Expected failure

def test_insert_large_dataset(setup_database):
    """Edge case: Insert large dataset"""
    large_docs = [{"_id": f"/documents/large_{i}.xml", "content": f"Content {i}"} for i in range(1000)]
    db.documents.insert_many(large_docs)
    count = db.documents.count_documents({"_id": {"$regex": "/documents/large_.*"}})
    assert count == 1000

def test_insert_invalid_structure(setup_database):
    """Error handling: Insert invalid document structure"""
    invalid_doc = {"_id": "/documents/invalid.xml", "invalid_field": "Invalid"}
    try:
        db.documents.insert_one(invalid_doc)
        inserted_doc = db.documents.find_one({"_id": "/documents/invalid.xml"})
        assert "content" not in inserted_doc
    except WriteError:
        assert True  # Expected failure

def test_unsupported_query(setup_database):
    """Error handling: Unsupported query"""
    try:
        db.documents.find({"unsupported": {"$foo": "bar"}})
        assert False  # Should not reach here
    except Exception:
        assert True  # Expected failure

if __name__ == "__main__":
    pytest.main()
