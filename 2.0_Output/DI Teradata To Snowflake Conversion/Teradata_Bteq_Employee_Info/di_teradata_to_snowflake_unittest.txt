import pytest
from snowflake.connector import connect

@pytest.fixture(scope="module")
def snowflake_connection():
    conn = connect(
        user='YOUR_USER',
        password='YOUR_PASSWORD',
        account='YOUR_ACCOUNT'
    )
    yield conn
    conn.close()

def setup_tables(cursor):
    cursor.execute("DROP TABLE IF EXISTS employee_bkup;")
    cursor.execute("DROP TABLE IF EXISTS Employee;")
    cursor.execute("DROP TABLE IF EXISTS Salary;")
    cursor.execute("""
        CREATE TEMP TABLE Employee (
            EmployeeNo INTEGER,
            FirstName VARCHAR(30),
            LastName VARCHAR(30),
            DepartmentNo SMALLINT
        );
    """)
    cursor.execute("""
        CREATE TEMP TABLE Salary (
            EmployeeNo INTEGER,
            NetPay INTEGER
        );
    """)
    cursor.execute("""
        CREATE TABLE employee_bkup (
            EmployeeNo INTEGER,
            FirstName VARCHAR(30),
            LastName VARCHAR(30),
            DepartmentNo SMALLINT,
            NetPay INTEGER
        );
    """)

def teardown_tables(cursor):
    cursor.execute("DROP TABLE IF EXISTS employee_bkup;")
    cursor.execute("DROP TABLE IF EXISTS Employee;")
    cursor.execute("DROP TABLE IF EXISTS Salary;")

def insert_and_backup(cursor):
    cursor.execute("""
        INSERT INTO employee_bkup
        SELECT 
            a.EmployeeNo, 
            a.FirstName, 
            a.LastName, 
            a.DepartmentNo, 
            b.NetPay
        FROM Employee a
        INNER JOIN Salary b
            ON a.EmployeeNo = b.EmployeeNo;
    """)

def test_happy_path(snowflake_connection):
    cursor = snowflake_connection.cursor()
    setup_tables(cursor)
    cursor.execute("INSERT INTO Employee VALUES (1, 'John', 'Doe', 10), (2, 'Jane', 'Smith', 20);")
    cursor.execute("INSERT INTO Salary VALUES (1, 5000), (2, 6000);")
    insert_and_backup(cursor)
    cursor.execute("SELECT * FROM employee_bkup ORDER BY EmployeeNo;")
    results = cursor.fetchall()
    assert results == [(1, 'John', 'Doe', 10, 5000), (2, 'Jane', 'Smith', 20, 6000)]
    teardown_tables(cursor)

    teardown_tables(cursor)
