-- Drop the backup table if it already exists
DROP TABLE IF EXISTS employee_bkup;

-- Create the backup table (optionally add UNIQUE constraint if needed)
CREATE OR REPLACE TABLE employee_bkup (
    EmployeeNo INTEGER,
    FirstName VARCHAR(30),
    LastName VARCHAR(30),
    DepartmentNo SMALLINT,
    NetPay INTEGER
);

-- Insert data into the backup table from Employee and Salary tables
INSERT INTO employee_bkup (
    EmployeeNo,
    FirstName,
    LastName,
    DepartmentNo,
    NetPay
)
SELECT
    a.EmployeeNo,
    a.FirstName,
    a.LastName,
    a.DepartmentNo,
    b.NetPay
FROM
    Employee a
    INNER JOIN Salary b
        ON a.EmployeeNo = b.EmployeeNo;
