-- Drop backup table if it exists for idempotency
DROP TABLE IF EXISTS employee_bkup;

-- Create backup table for employee data
CREATE TABLE employee_bkup (
    EmployeeNo INTEGER,
    FirstName VARCHAR(30),
    LastName VARCHAR(30),
    DepartmentNo SMALLINT,
    NetPay INTEGER,
    CONSTRAINT employee_bkup_unique_employee_no UNIQUE (EmployeeNo)
);

-- Insert joined employee and salary data into backup table
INSERT INTO employee_bkup (
    EmployeeNo,
    FirstName,
    LastName,
    DepartmentNo,
    NetPay
)
SELECT
    a.EmployeeNo,
    a.FirstName,
    a.LastName,
    a.DepartmentNo,
    b.NetPay
FROM
    Employee a
    INNER JOIN Salary b
        ON a.EmployeeNo = b.EmployeeNo;
