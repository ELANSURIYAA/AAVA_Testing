/*
 * This GWT Composite replicates a "Credit Summary" screen, showing a table of credit summary data.
 * The fields are based on the CRDT_SUMRY target definition in the XML.
 * The CSS provided at the end should be included in your module's CSS file for accurate styling.
 */

import com.google.gwt.core.client.GWT;
import com.google.gwt.user.client.ui.*;
import com.google.gwt.user.cellview.client.*;
import com.google.gwt.view.client.ListDataProvider;
import com.google.gwt.resources.client.ClientBundle;
import com.google.gwt.resources.client.CssResource;
import java.util.Arrays;
import java.util.List;

public class CreditSummaryScreen extends Composite {

    interface Resources extends ClientBundle {
        @Source("CreditSummaryScreen.css")
        Style style();
    }
    interface Style extends CssResource {
        String creditSummaryPanel();
        String creditSummaryHeader();
        String creditSummaryTable();
    }

    private static final Resources RESOURCES = GWT.create(Resources.class);

    public CreditSummaryScreen() {
        RESOURCES.style().ensureInjected();

        VerticalPanel mainPanel = new VerticalPanel();
        mainPanel.setStyleName(RESOURCES.style().creditSummaryPanel());

        Label header = new Label("Credit Summary");
        header.setStyleName(RESOURCES.style().creditSummaryHeader());
        mainPanel.add(header);

        // Table setup
        CellTable<CreditSummaryRow> table = new CellTable<>();
        table.setStyleName(RESOURCES.style().creditSummaryTable());

        // Define columns
        TextColumn<CreditSummaryRow> acctIdCol = new TextColumn<CreditSummaryRow>() {
            @Override
            public String getValue(CreditSummaryRow row) {
                return row.acctId;
            }
        };
        table.addColumn(acctIdCol, "Account ID");

        TextColumn<CreditSummaryRow> firstNameCol = new TextColumn<CreditSummaryRow>() {
            @Override
            public String getValue(CreditSummaryRow row) {
                return row.firstName;
            }
        };
        table.addColumn(firstNameCol, "First Name");

        TextColumn<CreditSummaryRow> lastNameCol = new TextColumn<CreditSummaryRow>() {
            @Override
            public String getValue(CreditSummaryRow row) {
                return row.lastName;
            }
        };
        table.addColumn(lastNameCol, "Last Name");

        TextColumn<CreditSummaryRow> balAmtCol = new TextColumn<CreditSummaryRow>() {
            @Override
            public String getValue(CreditSummaryRow row) {
                return row.balAmt;
            }
        };
        table.addColumn(balAmtCol, "Balance Amount");

        TextColumn<CreditSummaryRow> crdtScoreCol = new TextColumn<CreditSummaryRow>() {
            @Override
            public String getValue(CreditSummaryRow row) {
                return row.crdtScore;
            }
        };
        table.addColumn(crdtScoreCol, "Credit Score");

        // Sample data
        List<CreditSummaryRow> data = Arrays.asList(
            new CreditSummaryRow("1001", "John", "Doe", "1500.00", "720"),
            new CreditSummaryRow("1002", "Jane", "Smith", "2500.00", "680"),
            new CreditSummaryRow("1003", "Alice", "Brown", "3200.00", "750")
        );
        ListDataProvider<CreditSummaryRow> dataProvider = new ListDataProvider<>(data);
        dataProvider.addDataDisplay(table);

        mainPanel.add(table);

        initWidget(mainPanel);
    }

    // Data model for table rows
    public static class CreditSummaryRow {
        public String acctId;
        public String firstName;
        public String lastName;
        public String balAmt;
        public String crdtScore;

        public CreditSummaryRow(String acctId, String firstName, String lastName, String balAmt, String crdtScore) {
            this.acctId = acctId;
            this.firstName = firstName;
            this.lastName = lastName;
            this.balAmt = balAmt;
            this.crdtScore = crdtScore;
        }
    }
}
