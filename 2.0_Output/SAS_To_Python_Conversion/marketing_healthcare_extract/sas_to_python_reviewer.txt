# Python logic mimicking SAS flow
def marketing_extract_logic(hp_df, pn_df):
    df = hp_df.copy()
    # Trim and concatenate names
    df['member_full_name'] = df['member_first_name'].str.strip() + ' ' + df['member_last_name'].str.strip()
    # Age group logic
    df['age_group'] = np.where(
        (df['member_age'] >= 25) & (df['member_age'] <= 35), 'YOUNG_ADULT',
        np.where(
            (df['member_age'] >= 36) & (df['member_age'] <= 50), 'MID_AGE',
            np.where(
                (df['member_age'] > 50), 'SENIOR', 'UNKNOWN'
            )
        )
    )
    # Join provider network
    df = df.merge(pn_df, how='left', on='plan_id')
    # Marketing segment
    df['marketing_segment'] = df['plan_type'].str.upper() + '-' + df['region']
    # Campaign eligibility logic
    df['is_campaign_eligible'] = np.where(
        (df['plan_status'] == 'ACTIVE') &
        (df['member_age'].notnull()) &
        (df['member_age'] >= 25) & (df['member_age'] <= 60) &
        (df['plan_type'].isin(['PPO', 'HMO'])) &
        (df['region'].isin(['WEST', 'SOUTH'])) &
        (df['network_type'] == 'IN_NETWORK'),
        1, 0
    )
    # Filter for active status and non-null age (as in SAS WHERE clause)
    df = df[(df['plan_status'] == 'ACTIVE') & (df['member_age'].notnull())]
    return df

def final_campaign_list_logic(marketing_extract_df):
    return marketing_extract_df[marketing_extract_df['is_campaign_eligible'] == 1].copy()

def test_TC01_database_extraction(sample_healthcare_plans, sample_provider_network):
    df = marketing_extract_logic(sample_healthcare_plans, sample_provider_network)
    expected_cols = [
        'member_id', 'member_first_name', 'member_last_name', 'member_age', 'plan_id',
        'plan_type', 'region', 'plan_status', 'member_full_name', 'age_group',
        'network_type', 'marketing_segment', 'is_campaign_eligible'
    ]
    assert all(col in df.columns for col in expected_cols)
    assert len(df) == 5  # Only ACTIVE and non-null member_age
