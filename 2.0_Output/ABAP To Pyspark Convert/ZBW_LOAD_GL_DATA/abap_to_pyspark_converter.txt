from pyspark.sql import SparkSession from pyspark.sql.types import StructType, StructField, StringType, DoubleType, DateType from pyspark.sql.functions import col, lit, to_date

Initialize Spark Session
spark = SparkSession.builder.appName("FinanceDataLoadToBW").getOrCreate()

Define schema equivalent to zbw_finance_data
schema = StructType([ StructField("bukrs", StringType(), True), # Company Code StructField("fiscyear", StringType(), True), # Fiscal Year StructField("costcenter", StringType(), True), # Cost Center StructField("gl_account", StringType(), True), # GL Account StructField("amount", StringType(), True), # Transaction Amount (could be DoubleType if numeric) StructField("currency", StringType(), True), # Currency StructField("posting_date", StringType(), True) # Posting Date (could be DateType if parsing) ])

File path (update as needed)
input_file = "/usr/sap/interfaces/finance_data.csv"

Read CSV file into DataFrame
Assuming the file does not have a header and fields are comma-separated
df_raw = spark.read.csv(input_file, schema=schema, header=False)

Filter out rows that do not have exactly 7 fields
df_valid = df_raw.filter( col("bukrs").isNotNull() & col("fiscyear").isNotNull() & col("costcenter").isNotNull() & col("gl_account").isNotNull() & col("amount").isNotNull() & col("currency").isNotNull() & col("posting_date").isNotNull() )

Optionally, parse amount and posting_date if needed
df_valid = df_valid.withColumn("amount", col("amount").cast(DoubleType()))
df_valid = df_valid.withColumn("posting_date", to_date(col("posting_date"), "yyyy-MM-dd"))
Handle incorrect format lines (log or write to error file)
df_invalid = df_raw.filter( col("bukrs").isNull() | col("fiscyear").isNull() | col("costcenter").isNull() | col("gl_account").isNull() | col("amount").isNull() | col("currency").isNull() | col("posting_date").isNull() ) if df_invalid.count() > 0: df_invalid.write.mode("overwrite").csv("/usr/sap/interfaces/finance_data_error_lines.csv") print("Error: Incorrect file format in some lines. See finance_data_error_lines.csv for details.")

Insert data into SAP BW Table
