import pytest
from pyspark.sql import SparkSession
from pyspark.sql.types import StructType, StructField, StringType, DecimalType, DateType
from pyspark.sql.utils import AnalysisException

@pytest.fixture(scope="module")
def spark():
    return SparkSession.builder.master("local[1]").appName("ABAPtoPySparkTest").getOrCreate()

# Test data sample (simulate CSV file lines)
sample_csv = [
    "1000,2023,CC01,GL123,1000.50,USD,2023-01-01",
    "2000,2022,CC02,GL456,200.75,EUR,2022-12-31",
    "BAD,LINE,WITH,TOO,FEW,FIELDS"
]

expected_data = [
    ("1000", "2023", "CC01", "GL123", "1000.50", "USD", "2023-01-01"),
    ("2000", "2022", "CC02", "GL456", "200.75", "EUR", "2022-12-31")
]

schema = StructType([
    StructField("bukrs", StringType(), True),
    StructField("fiscyear", StringType(), True),
    StructField("costcenter", StringType(), True),
    StructField("gl_account", StringType(), True),
    StructField("amount", StringType(), True),  # Could be DecimalType()
    StructField("currency", StringType(), True),
    StructField("posting_date", StringType(), True),  # Could be DateType()
])

def create_test_csv(tmp_path, lines):
    file_path = tmp_path / "finance_data.csv"
    with open(file_path, "w") as f:
        for line in lines:
            f.write(line + "
")
    return str(file_path)

def read_and_transform_csv(spark, file_path):
    # Simulate the PySpark conversion logic
    df = spark.read.csv(file_path, schema=schema, header=False)
    # Filter out rows with incorrect field count
    df = df.filter(df["posting_date"].isNotNull())
    return df

def test_TC01_file_handling_conversion(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    try:
        df = spark.read.csv(file_path, schema=schema, header=False)
        assert df.count() > 0
    except Exception as e:
        pytest.fail(f"File handling failed: {e}")

def test_TC02_data_transformation(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = read_and_transform_csv(spark, file_path)
    rows = df.collect()
    assert rows[0]["bukrs"] == "1000"
    assert rows[1]["gl_account"] == "GL456"

def test_TC03_error_handling_sy_subrc(spark, tmp_path):
    # Try reading a non-existent file
    with pytest.raises(AnalysisException):
        spark.read.csv("non_existent_file.csv", schema=schema, header=False).collect()

def test_TC04_internal_table_append(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = read_and_transform_csv(spark, file_path)
    assert df.count() == 2  # Only valid lines are loaded

def test_TC05_dml_operation_insert(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = read_and_transform_csv(spark, file_path)
    output_path = tmp_path / "output"
    df.write.mode("overwrite").csv(str(output_path))
    # Read back and check
    df2 = spark.read.csv(str(output_path), schema=schema, header=False)
    assert df2.count() == 2

def test_TC06_conditional_logic_field_count(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = spark.read.csv(file_path, schema=schema, header=False)
    # The bad line should have posting_date as None
    bad_rows = df.filter(df["posting_date"].isNull()).collect()
    assert len(bad_rows) == 1

def test_TC07_data_consistency(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = read_and_transform_csv(spark, file_path)
    actual = [tuple(row) for row in df.collect()]
    assert actual == expected_data

def test_TC08_date_and_numeric_type_handling(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = read_and_transform_csv(spark, file_path)
    # Check date and amount formats
    for row in df.collect():
        assert "-" in row["posting_date"]
        float(row["amount"])  # Should not raise

def test_TC09_performance_optimization(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = read_and_transform_csv(spark, file_path)
    # Test repartition (optimization)
    df2 = df.repartition(2)
    assert df2.rdd.getNumPartitions() == 2

def test_TC10_edge_case_incorrect_format(spark, tmp_path):
    file_path = create_test_csv(tmp_path, sample_csv)
    df = spark.read.csv(file_path, schema=schema, header=False)
    # The bad line should be detected
    bad_rows = df.filter(df["posting_date"].isNull()).collect()
    assert len(bad_rows) == 1
