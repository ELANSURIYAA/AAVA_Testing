import pytest
from pyspark.sql import SparkSession, Row
from pyspark.sql.utils import AnalysisException
import os

# GCP DataProc-optimized SparkSession fixture
@pytest.fixture(scope="module")
def spark():
    spark = SparkSession.builder \
        .appName("pytest-gcp-dataproc") \
        .master("local[2]") \
        .getOrCreate()
    yield spark
    spark.stop()

# Helper function to create mock DataFrame
def create_mock_df(spark, data, schema=None):
    return spark.createDataFrame(data, schema=schema)

# Mock BigQuery writer (to avoid actual writes in unit tests)
class MockDataFrameWriter:
    def __init__(self, df):
        self.df = df
        self.options = {}
        self.format_type = None

    def format(self, format_type):
        self.format_type = format_type
        return self

    def option(self, key, value):
        self.options[key] = value
        return self

    def save(self):
        # Instead of writing, just return the DataFrame and options for assertion
        return (self.df, self.format_type, self.options)

# Patch DataFrame.write to use the mock writer
@pytest.fixture(autouse=True)
def patch_write(monkeypatch):
    monkeypatch.setattr("pyspark.sql.dataframe.DataFrame.write", property(lambda self: MockDataFrameWriter(self)))

# Test cases

def test_person_extraction_happy_path(spark):
    data = [Row(PersonID=1, DOB="1990-01-01", Gender="M", Race="Asian", Ethnicity="Not Hispanic", SourceSystem="sys1", IngestionTimestamp="2024-01-01T00:00:00Z")]
    df = create_mock_df(spark, data)
    person_df = df.select(
        "PersonID", "DOB", "Gender", "Race", "Ethnicity", "SourceSystem", "IngestionTimestamp"
    )
    assert set(person_df.columns) == {"PersonID", "DOB", "Gender", "Race", "Ethnicity", "SourceSystem", "IngestionTimestamp"}
    assert person_df.count() == 1

def test_contactinfo_extraction_happy_path(spark):
    data = [Row(PersonID=2, ContactInfo=Row(Address="123 St", Phone="1234567890", Email="test@example.com"))]
    df = create_mock_df(spark, data)
    contact_info_df = df.select(
        "PersonID",
        "ContactInfo.Address",
        "ContactInfo.Phone",
        "ContactInfo.Email"
    )
    row = contact_info_df.collect()[0]
    assert row.Address == "123 St"
    assert row.Phone == "1234567890"
    assert row.Email == "test@example.com"

def test_language_extraction_happy_path(spark):
    data = [Row(PersonID=3, Language=Row(Spoken="English", Written="French"))]
    df = create_mock_df(spark, data)
    language_df = df.select(
        "PersonID",
        "Language.Spoken",
        "Language.Written"
    )
    row = language_df.collect()[0]
    assert row.Spoken == "English"
    assert row.Written == "French"

def test_empty_input(spark):
    df = create_mock_df(spark, [])
    person_df = df.select("PersonID") if "PersonID" in df.columns else df
    assert person_df.count() == 0

def test_null_nested_fields(spark):
    data = [Row(PersonID=6, ContactInfo=None, Language=None)]
    df = create_mock_df(spark, data)
    contact_info_df = df.select(
        "PersonID",
        "ContactInfo.Address",
        "ContactInfo.Phone",
        "ContactInfo.Email"
    )
    language_df = df.select(
        "PersonID",
        "Language.Spoken",
        "Language.Written"
    )
    row_c = contact_info_df.collect()[0]
    row_l = language_df.collect()[0]
    assert row_c.Address is None and row_c.Phone is None and row_c.Email is None
    assert row_l.Spoken is None and row_l.Written is None

def test_schema_mismatch(spark):
    data = [Row(PersonID=7, ContactInfo="not_a_struct")]
    df = create_mock_df(spark, data)
    with pytest.raises(AnalysisException):
        df.select("ContactInfo.Address").collect()

def test_invalid_json_input(spark):

    with pytest.raises(Exception):

def test_bigquery_write_options(spark):
    data = [Row(PersonID=8)]
    df = create_mock_df(spark, data)
    writer = MockDataFrameWriter(df)
    result = writer.format("bigquery").option("table", "project.dataset.Person").save()
    assert result[1] == "bigquery"
    assert result[2]["table"] == "project.dataset.Person"

def test_performance_large_input(spark):
    # Simulate large input by repeating rows (not a true perf test, but checks logic)
    data = [Row(PersonID=i) for i in range(10000)]
    df = create_mock_df(spark, data)
    person_df = df.select("PersonID")
    assert person_df.count() == 10000

def test_multiple_records(spark):
    data = [
        Row(PersonID=9, ContactInfo=Row(Address="A", Phone="1", Email="a@a.com"), Language=Row(Spoken="E", Written="F")),
        Row(PersonID=10, ContactInfo=Row(Address="B", Phone="2", Email="b@b.com"), Language=Row(Spoken="S", Written="G"))
    ]
    df = create_mock_df(spark, data)
    assert df.count() == 2
    contact_info_df = df.select("ContactInfo.Address")
    assert contact_info_df.count() == 2

def test_unexpected_fields_ignored(spark):
    data = [Row(PersonID=11, ExtraField="ignore_me")]
    df = create_mock_df(spark, data)
    person_df = df.select("PersonID")
    assert "PersonID" in person_df.columns
    assert "ExtraField" not in person_df.columns
