def run_customer_orders_query(spark, orders, users, order_items, products):
    # Simulate DBT ref macro by using DataFrames directly
    o = orders.alias("o")
    u = users.alias("u")
    oi = order_items.alias("oi")
    p = products.alias("p")

    customer_orders = (
        o.join(u, o.user_id == u.id, "inner")
         .join(oi, o.id == oi.order_id, "inner")
         .join(p, oi.product_id == p.id, "inner")
         .filter(o.status == "completed")
         .select(
            o.id.alias("order_id"),
            o.user_id,
            u.name.alias("customer_name"),
            u.email.alias("customer_email"),
            o.total_amount,
            o.created_at.alias("order_date"),
            p.category.alias("product_category"),
            p.price.alias("product_price")
         )
    )

    order_summary = (
        customer_orders.groupBy("user_id", "customer_name", "customer_email")
        .agg(
            F.count("order_id").alias("total_orders"),
            F.sum("total_amount").alias("total_spent"),
            F.avg("total_amount").alias("avg_order_value"),
            F.collect_set("product_category").alias("categories_bought")
        )
        .orderBy(F.col("total_spent").desc())
    )
    return order_summary

# TC01: Happy path
def test_happy_path(spark_session):
    """TC01: Happy path with multiple users and products"""
    orders, users, order_items, products = create_test_data(spark_session)
    result = run_customer_orders_query(spark_session, orders, users, order_items, products)
    rows = result.collect()
    assert len(rows) == 4  # Only users with completed orders
    # Check Charlie (user_id=104) has one order, correct spent
    charlie = [r for r in rows if r.user_id == 104][0]
    assert charlie.total_orders == 1
    assert charlie.total_spent == 300.0
    assert charlie.avg_order_value == 300.0
    assert "Clothing" in charlie.categories_bought

def test_performance_large_dataset(spark_session):
    """TC08: Performance: Large dataset executes within reasonable time"""
    orders, users, order_items, products = create_test_data(spark_session)
    # Expand orders to 10000 rows
    large_orders = orders
    for i in range(1000):
        large_orders = large_orders.union(orders.withColumn("id", F.col("id") + (i+1)*10))
    import time
    start = time.time()
    result = run_customer_orders_query(spark_session, large_orders, users, order_items, products)
    elapsed = time.time() - start
    assert elapsed < 10  # Should run in under 10 seconds

def test_dbt_syntax_compliance():
    """TC10: DBT syntax: Ref macros and aggregations compile"""
    # This test is a placeholder: actual DBT compilation is checked in DBT environment
    dbt_sql = """
    WITH customer_orders AS (
        SELECT
            o.id AS order_id,
            o.user_id,
            u.name AS customer_name,
            u.email AS customer_email,
            o.total_amount,
            o.created_at AS order_date,
            p.category AS product_category,
            p.price AS product_price
        FROM {{ ref('orders') }} o
        JOIN {{ ref('users') }} u ON o.user_id = u.id
        JOIN {{ ref('order_items') }} oi ON o.id = oi.order_id
        JOIN {{ ref('products') }} p ON oi.product_id = p.id
        WHERE o.status = 'completed'
    ),
    order_summary AS (
        SELECT 
            user_id,
            customer_name,
            customer_email,
            COUNT(order_id) AS total_orders,
            SUM(total_amount) AS total_spent,
            AVG(total_amount) AS avg_order_value,
            ARRAY_AGG(DISTINCT product_category) AS categories_bought
        FROM customer_orders
        GROUP BY user_id, customer_name, customer_email
    )
    SELECT * FROM order_summary
    ORDER BY total_spent DESC;
    """
