1. Physical Layer DDL Script:

-- Member Table (Main Entity, excluding nested ContactInfo and Language columns)
CREATE TABLE IF NOT EXISTS Member (
  PersonID STRING,
  DOB DATE,
  Gender STRING,
  Race STRING,
  Ethnicity STRING,
  SourceSystem STRING,
  IngestionTimestamp TIMESTAMP
);

-- ContactInfo Table
CREATE TABLE IF NOT EXISTS ContactInfo (
  Address STRING,
  Phone STRING,
  Email STRING
);

-- Language Table
CREATE TABLE IF NOT EXISTS Language (
  Spoken ARRAY<STRING>,
  Written STRING
);
