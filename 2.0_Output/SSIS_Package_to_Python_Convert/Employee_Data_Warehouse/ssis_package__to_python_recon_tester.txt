def execute_ssis_package(ssis_package_path: str, batch_id: str, log_file: str) -> Dict[str, Any]: """ Execute SSIS package using DTExec and capture logs. """ log_and_print(f"Executing SSIS package: {ssis_package_path}") dtexec_cmd = [ "dtexec", "/F", ssis_package_path, "/SET", f"\Package.Variables[User::BatchID].Properties[Value];{batch_id}", "/SET", f"\Package.Variables[User::LogFilePath].Properties[Value];{log_file}", "/REPORTING", "E" ] try: result = subprocess.run(dtexec_cmd, capture_output=True, text=True, check=False) log_and_print(f"SSIS STDOUT: {result.stdout}") log_and_print(f"SSIS STDERR: {result.stderr}") return { "returncode": result.returncode, "stdout": result.stdout, "stderr": result.stderr } except Exception as ex: log_and_print(f"Error executing SSIS package: {ex}") return { "returncode": -1, "stdout": "", "stderr": str(ex) }

def extract_table_to_parquet(engine, table_name, output_dir, prefix): """Extract table to Parquet file.""" df = pd.read_sql_table(table_name, engine) out_path = os.path.join(output_dir, f"{prefix}{table_name}{BATCH_ID}.parquet") df.to_parquet(out_path, index=False) log_and_print(f"Extracted {table_name} to {out_path}") return out_path

def extract_ssis_outputs(): """ Extract all relevant SSIS output tables to Parquet. def execute_python_etl(script_path: str, batch_id: str, log_file: str) -> Dict[str, Any]: """ Execute the Python ETL script and capture logs. """ log_and_print(f"Executing Python ETL script: {script_path}") env = os.environ.copy() env["BATCH_ID"] = batch_id env["LOG_FILE_PATH"] = log_file try: result = subprocess.run( [sys.executable, script_path], capture_output=True, text=True, env=env, check=False ) log_and_print(f"Python ETL STDOUT: {result.stdout}") log_and_print(f"Python ETL STDERR: {result.stderr}") return { "returncode": result.returncode, "stdout": result.stdout, "stderr": result.stderr } except Exception as ex: log_and_print(f"Error executing Python ETL: {ex}") return { "returncode": -1, "stdout": "", "stderr": str(ex) }

5. DATA COMPARISON LOGIC

def compare_dataframes(df1, df2, key_columns=None): """ Compare two dataframes and return match status, discrepancies, and sample mismatches. """ result = { "row_count_ssis": len(df1), "row_count_python": len(df2), "row_count_match": len(df1) == len(df2), "column_match": list(df1.columns) == list(df2.columns), "column_discrepancies": [], "data_mismatches": [] } # Compare columns if list(df1.columns) != list(df2.columns): result["column_discrepancies"] = list(set(df1.columns).symmetric_difference(set(df2.columns))) # Compare data if result["row_count_match"] and result["column_match"]: mismatches = [] for idx, (row1, row2) in enumerate(zip(df1.itertuples(index=False), df2.itertuples(index=False))): if row1 != row2: mismatches.append({"row": idx, "ssis": row1, "python": row2}) if len(mismatches) >= 5: break result["data_mismatches"] = mismatches result["match_percentage"] = 100.0 * (len(df1) - len(mismatches)) / len(df1) if len(df1) else 100.0 if not mismatches: result["match_status"] = "MATCH" elif len(mismatches) < len(df1) * 0.05: result["match_status"] = "PARTIAL MATCH" else: result["match_status"] = "NO MATCH" else: result["match_status"] = "NO MATCH" result["match_percentage"] = 0.0 return result

def compare_outputs(ssis_outputs, python_outputs): """ Compare all outputs and return a report. """ report = {} for table in ssis_outputs: ssis_file = ssis_outputs[table] py_file = python_outputs[table] df_ssis = pd.read_parquet(ssis_file) df_py = pd.read_parquet(py_file) # Handle possible data type differences try: df_ssis = df_ssis.astype(df_py.dtypes.to_dict()) except Exception: pass comp = compare_dataframes(df_ssis, df_py) report[table] = comp return report

def validate_control_flow(ssis_log_path, python_log_path): """ Compare control flow and error handling between SSIS and Python logs. """ def parse_log(log_path): if not os.path.exists(log_path): return [] with open(log_path, "r") as f: return f.readlines() ssis_log = parse_log(ssis_log_path) python_log = parse_log(python_log_path) # Simple comparison: check for main events control_flow_match = all(event in python_log for event in ssis_log if "Package Execution" in event or "Error" in event) return { "ssis_log_events": ssis_log[-10:], # last 10 events "python_log_events": python_log[-10:], "control_flow_match": control_flow_match }
