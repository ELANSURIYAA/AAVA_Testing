import pytest
from unittest.mock import MagicMock, patch, mock_open
from datetime import datetime

# Assume these are the Python equivalents of the SSIS script tasks

class DTSExecResult:
    Success = 0
    Failure = 1

class DtsMock:
    def __init__(self, variables):
        self.Variables = variables
        self.TaskResult = None
        self.Events = MagicMock()

def log_execution(Dts):
    try:
        logFilePath = Dts.Variables["LogFilePath"].Value
        batchID = Dts.Variables.get("BatchID", MagicMock(Value="")).Value
        with open(logFilePath, "a") as sw:
            sw.write(f"{datetime.now()} - Package Execution Started. Batch ID: {batchID}
")
        Dts.TaskResult = DTSExecResult.Success
    except Exception as ex:
        Dts.Events.FireError(0, "Log Execution", str(ex), "", 0)
        Dts.TaskResult = DTSExecResult.Failure

def error_logging(Dts):
    try:
        connString = Dts.Variables["DestinationDBConnection"].Value
        errorMsg = Dts.Variables.get("System::ErrorDescription", MagicMock(Value=None)).Value
        batchID = Dts.Variables.get("BatchID", MagicMock(Value=None)).Value
        # Simulate DB connection and insert
        with patch("pyodbc.connect") as mock_conn:
            conn = mock_conn(connString)
            cursor = conn.cursor()
            cursor.execute(
                "INSERT INTO SSIS_Error_Log (ErrorMessage, ErrorTime, BatchID) VALUES (?, GETDATE(), ?)",
                (errorMsg, batchID)
            )
            conn.commit()
        Dts.TaskResult = DTSExecResult.Failure
    except Exception as ex:
        Dts.Events.FireError(0, "Error Logging", str(ex), "", 0)
        Dts.TaskResult = DTSExecResult.Failure

@pytest.fixture
def dts_mock():
    class Variable:
        def __init__(self, value):
            self.Value = value
    return lambda vars: DtsMock({k: Variable(v) for k, v in vars.items()})

# TC01: Log execution with valid LogFilePath and BatchID
def test_log_execution_success(dts_mock):
    dts = dts_mock({"LogFilePath": "/tmp/test_log.txt", "BatchID": "123"})
    with patch("builtins.open", mock_open()) as m:
        log_execution(dts)
        m().write.assert_called()
        assert dts.TaskResult == DTSExecResult.Success

# TC02: Log execution with missing LogFilePath variable
def test_log_execution_missing_logfile(dts_mock):
    dts = dts_mock({"BatchID": "123"})
    log_execution(dts)
    dts.Events.FireError.assert_called()
    assert dts.TaskResult == DTSExecResult.Failure

# TC04: Log execution with inaccessible log file path
def test_log_execution_inaccessible_logfile(dts_mock):
    dts = dts_mock({"LogFilePath": "/root/forbidden.txt", "BatchID": "123"})
    with patch("builtins.open", side_effect=PermissionError("Permission denied")):
        log_execution(dts)
        dts.Events.FireError.assert_called()
        assert dts.TaskResult == DTSExecResult.Failure

# TC05: Error logging with valid DB connection, ErrorDescription, and BatchID
def test_error_logging_success(dts_mock):
