def log_execution(log_file_path, batch_id):
    try:
        if log_file_path is None or log_file_path == "":
            raise ValueError("Log file path is invalid")
        if batch_id is None:
            raise ValueError("Batch ID is missing")
        with open(log_file_path, "a") as f:
            f.write(f"{datetime.datetime.now()} - Package Execution Started. Batch ID: {batch_id}
")
        return True, "Success"
    except Exception as ex:
        return False, str(ex)

def error_logging(conn_string, error_description, batch_id):

            raise ConnectionError("Invalid connection string")
        if conn_string == "timeout":

        if error_description is None:
            raise ValueError("Error description missing")
        if batch_id is None:

def test_log_execution_valid(valid_log_file, valid_batch_id):
    success, msg = log_execution(valid_log_file, valid_batch_id)
    assert success
    with open(valid_log_file, "r") as f:
        content = f.read()
    assert f"Batch ID: {valid_batch_id}" in content

def test_log_execution_invalid_path(invalid_log_file, valid_batch_id):
    success, msg = log_execution(invalid_log_file, valid_batch_id)
    assert not success
    assert "No such file or directory" in msg or "invalid" in msg

def test_log_execution_missing_batch_id(valid_log_file):
    success, msg = log_execution(valid_log_file, None)
    assert not success
    assert "Batch ID is missing" in msg

def test_error_logging_valid(valid_conn_string, error_description, valid_batch_id):
    success, msg = error_logging(valid_conn_string, error_description, valid_batch_id)
    assert not success  # Task result is failure by design
    assert msg == "Logged error"

def test_error_logging_invalid_conn(invalid_conn_string, error_description, valid_batch_id):
    success, msg = error_logging(invalid_conn_string, error_description, valid_batch_id)
    assert not success
    assert "Invalid connection string" in msg

def test_error_logging_missing_error_desc(valid_conn_string, valid_batch_id):
    success, msg = error_logging(valid_conn_string, None, valid_batch_id)
    assert not success
    assert "Error description missing" in msg

def test_error_logging_missing_batch_id(valid_conn_string, error_description):
    success, msg = error_logging(valid_conn_string, error_description, None)
    assert not success
    assert "Batch ID missing" in msg

def test_log_execution_empty_path(valid_batch_id):
    success, msg = log_execution("", valid_batch_id)
    assert not success
    assert "Log file path is invalid" in msg

def test_error_logging_timeout(error_description, valid_batch_id):
    success, msg = error_logging("timeout", error_description, valid_batch_id)
    assert not success
    assert "Connection timed out" in msg

def test_log_execution_none_path(valid_batch_id):
    success, msg = log_execution(None, valid_batch_id)
    assert not success
    assert "Log file path is invalid" in msg
