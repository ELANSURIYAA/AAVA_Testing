-- BigQuery SQL Script: Sales Fact Table Load with Data Quality Validation and Audit Logging

CREATE OR REPLACE PROCEDURE dw.sp_load_sales_fact()
BEGIN
  -- Variable Declarations
  DECLARE batch_id STRING DEFAULT GENERATE_UUID();
  DECLARE start_time DATETIME DEFAULT CURRENT_DATETIME();
  DECLARE end_time DATETIME;
  DECLARE rows_inserted INT64 DEFAULT 0;
  DECLARE rows_rejected INT64 DEFAULT 0;
  DECLARE error_message STRING;
  DECLARE proc_name STRING DEFAULT 'sp_load_sales_fact';

  -- Start Audit Logging
  INSERT INTO dw.Audit_Log (
    Batch_ID,
    Procedure_Name,
    Start_Time,
    Status,
    Message
  )
  VALUES (
    batch_id,
    proc_name,
    start_time,
    'STARTED',
    'Sales Fact Load Initiated'
  );

  -- Data Quality Validation: Identify Invalid Rows
  WITH invalid_customer AS (
    SELECT Transaction_ID, 'Missing CustomerID' AS Reason
    FROM stg.Sales_Transactions
    WHERE Customer_ID IS NULL
  ),
  invalid_quantity AS (
    SELECT Transaction_ID, 'Invalid Quantity' AS Reason
    FROM stg.Sales_Transactions
    WHERE Quantity <= 0
  ),
  InvalidRows AS (
    SELECT * FROM invalid_customer
    UNION ALL
    SELECT * FROM invalid_quantity
  )

  -- Remove Invalid Rows from Staging and Count Rejected
  , Clean_Staging AS (
    SELECT *
    FROM stg.Sales_Transactions s
    LEFT JOIN InvalidRows i
      ON s.Transaction_ID = i.Transaction_ID
    WHERE i.Transaction_ID IS NULL
  )

  -- Transform Cleaned Data for Fact Table Load
  , transformed AS (
    SELECT
      s.Transaction_ID,
      s.Customer_ID,
      s.Product_ID,
      s.Sales_Date,
      s.Quantity,
      s.Unit_Price,
      s.Quantity * s.Unit_Price AS Total_Sales_Amount,
      d.Region_ID,
      c.Customer_Segment,
      CURRENT_DATETIME() AS Load_Timestamp,
      batch_id AS Batch_ID
    FROM Clean_Staging s
    INNER JOIN dw.Dim_Customer c
      ON s.Customer_ID = c.Customer_ID
    INNER JOIN dw.Dim_Date d
      ON DATE(s.Sales_Date) = d.Date_Value
  )

  -- Insert Cleaned and Transformed Data into Fact Table
  INSERT INTO dw.Fact_Sales (
    Transaction_ID,
    Customer_ID,
    Product_ID,
    Sales_Date,
    Quantity,
    Unit_Price,
    Total_Sales_Amount,
    Region_ID,
    Customer_Segment,
    Load_Timestamp,
    Batch_ID
  )
  SELECT
    Transaction_ID,
    Customer_ID,
    Product_ID,
    Sales_Date,
    Quantity,
    Unit_Price,
    Total_Sales_Amount,
    Region_ID,
    Customer_Segment,
    Load_Timestamp,
    Batch_ID
  FROM transformed;

  -- Track Inserted Row Count
  SET rows_inserted = (SELECT COUNT(*) FROM transformed);

  -- Track Rejected Row Count
  SET rows_rejected = (
    SELECT COUNT(*) FROM (
      SELECT Transaction_ID FROM invalid_customer
      UNION ALL
      SELECT Transaction_ID FROM invalid_quantity
    )
  );

  -- Truncate Staging Table (delete all rows)
  DELETE FROM stg.Sales_Transactions WHERE TRUE;

  -- Log Validation Failures
  INSERT INTO dw.DQ_Failures (
    Transaction_ID,
    Failure_Reason,
    Logged_Timestamp,
    Batch_ID
  )
  SELECT
    Transaction_ID,
    Reason,
    CURRENT_DATETIME(),
    batch_id
  FROM InvalidRows;

  -- End Audit Log
  SET end_time = CURRENT_DATETIME();

  UPDATE dw.Audit_Log
  SET
    End_Time = end_time,
    Rows_Inserted = rows_inserted,
    Rows_Rejected = rows_rejected,
    Status = 'COMPLETED',
    Message = CONCAT('Inserted ', CAST(rows_inserted AS STRING), ' rows; Rejected ', CAST(rows_rejected AS STRING), ' rows.')
  WHERE Batch_ID = batch_id;

EXCEPTION WHEN ERROR THEN
  SET end_time = CURRENT_DATETIME();
  SET error_message = ERROR_MESSAGE();

  UPDATE dw.Audit_Log
  SET
    End_Time = end_time,
    Status = 'FAILED',
    Message = error_message
  WHERE Batch_ID = batch_id;

END;

-- Notes:
-- - All table references are generic (dw.Fact_Sales, dw.Audit_Log, dw.DQ_Failures, stg.Sales_Transactions, dw.Dim_Customer, dw.Dim_Date).
-- - Data quality validation logic is implemented using CTEs for modularity and performance.
-- - Audit logging and error handling follow BigQuery best practices.
-- - Use partitioned and clustered tables for optimal performance in production.
-- - Ensure all referenced tables exist and have compatible schemas.
