def compare_outputs(df_pg, df_ps):
    # Compare DataFrames for equality (ignoring order)
    df_pg_sorted = df_pg.sort_values(by=df_pg.columns.tolist()).reset_index(drop=True)
    df_ps_sorted = df_ps.sort_values(by=df_ps.columns.tolist()).reset_index(drop=True)
    comparison = df_pg_sorted.equals(df_ps_sorted)
    diff_pg = pd.concat([df_pg_sorted, df_ps_sorted]).drop_duplicates(keep=False)
    diff_ps = pd.concat([df_ps_sorted, df_pg_sorted]).drop_duplicates(keep=False)
    return comparison, diff_pg, diff_ps
