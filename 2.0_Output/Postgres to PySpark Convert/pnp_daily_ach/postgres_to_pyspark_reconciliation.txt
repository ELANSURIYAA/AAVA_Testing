def compare_outputs(pg_output, pyspark_output):
    # Compare DataFrames for inserts, updates, deletes
    pg_df = pg_output
    ps_df = pyspark_output

    # Matching records
    matching = pg_df.join(ps_df, on="transaction_par_nbr", how="inner")
    # Not matching records
    not_matching_pg = pg_df.join(ps_df, on="transaction_par_nbr", how="left_anti")
    not_matching_ps = ps_df.join(pg_df, on="transaction_par_nbr", how="left_anti")

    return matching, not_matching_pg, not_matching_ps

def generate_report(matching, not_matching_pg, not_matching_ps):
    report = {
        "matching_count": matching.count(),
        "not_matching_pg_count": not_matching_pg.count(),
        "not_matching_ps_count": not_matching_ps.count(),
        "matching_sample": matching.limit(5).toPandas().to_dict(orient="records"),
        "not_matching_pg_sample": not_matching_pg.limit(5).toPandas().to_dict(orient="records"),
        "not_matching_ps_sample": not_matching_ps.limit(5).toPandas().to_dict(orient="records"),
    }
    logger.info(f"Reconciliation Report: {report}")
    return report

@pytest.mark.parametrize("test_case_id", ["TC01", "TC02", "TC03", "TC04", "TC05"])
def test_reconciliation(spark, test_case_id):
    logger.info(f"Running reconciliation test for {test_case_id}")

    # Load outputs
    pg_output = load_postgres_output(test_case_id)
    pyspark_output = load_pyspark_output(spark, test_case_id)

    # Compare outputs
    matching, not_matching_pg, not_matching_ps = compare_outputs(pg_output, pyspark_output)

    # Generate report
    report = generate_report(matching, not_matching_pg, not_matching_ps)

    # Assertions
    assert report["not_matching_pg_count"] == 0, f"PostgreSQL output has unmatched records for {test_case_id}"
    assert report["not_matching_ps_count"] == 0, f"PySpark output has unmatched records for {test_case_id}"
    assert report["matching_count"] > 0, f"No matching records found for {test_case_id}"

    logger.info(f"Test {test_case_id} passed.")
