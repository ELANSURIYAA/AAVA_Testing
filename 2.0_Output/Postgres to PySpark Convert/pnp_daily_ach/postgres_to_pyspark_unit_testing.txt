def test_TC05_person_regex(spark):
    df = spark.createDataFrame([
        {"zxx__cust_name_mod_new": "JOHN A DOE", "entity_type": "PERSON-TBR"}
    ])
    person_patterns = [r"^[A-Z]{3,} [A-Z] [A-Z]{2,}$"]
    person_regex = "|".join(person_patterns)
    df2 = df.withColumn("entity_type", F.when(
        (F.col("entity_type") == "PERSON-TBR") & (F.col("zxx__cust_name_mod_new").rlike(person_regex)), "PERSON"
    ).otherwise(F.col("entity_type")))
    assert df2.collect()[0]["entity_type"] == "PERSON"

def test_TC06_business_regex(spark):
    df = spark.createDataFrame([
        {"zxx__cust_name_mod": "ACME LLC", "zxx__cust_name_mod_new": "ACME LLC", "entity_type": "PERSON"}
    ])
    business_patterns = [r"(^| )(LLC)( |$)"]
    business_regex = "|".join(business_patterns)
    df2 = df.withColumn("entity_type", F.when(
        (F.col("entity_type") != "BUSINESS") & (F.col("zxx__cust_name_mod_new").rlike(business_regex)), "BUSINESS"
    ).otherwise(F.col("entity_type")))
    assert df2.collect()[0]["entity_type"] == "BUSINESS"

def test_TC07_business_only(spark):
    df = spark.createDataFrame([
        {"entity_type": "PERSON", "business_only": 1}
    ])
    df2 = df.withColumn("entity_type", F.when(
        (F.col("entity_type") != "BUSINESS") & (F.col("business_only") > 0), "BUSINESS"
    ).otherwise(F.col("entity_type")))
    assert df2.collect()[0]["entity_type"] == "BUSINESS"

def test_TC08_unidentifiable(spark):
    df = spark.createDataFrame([
        {"zxx__cust_name_mod": "A", "entity_type": "PERSON"}
    ])
    df2 = df.withColumn("entity_type", F.when(
        F.length(F.regexp_replace(F.col("zxx__cust_name_mod"), "[^A-Z]", "")) <= 2, "UNIDENTIFIABLE"
    ).otherwise(F.col("entity_type")))
    assert df2.collect()[0]["entity_type"] == "UNIDENTIFIABLE"

def test_TC10_window_row_number(spark):
    df = spark.createDataFrame([
        {"transaction_par_nbr": "T1", "business_dt": "2024-06-01", "destination_pnp_cde": "PERSON"},
        {"transaction_par_nbr": "T1", "business_dt": "2024-06-01", "destination_pnp_cde": "BUSINESS"}
    ])
    window_spec = Window.partitionBy("transaction_par_nbr", "business_dt").orderBy("destination_pnp_cde")
    df2 = df.withColumn("r_id", F.row_number().over(window_spec)).filter(F.col("r_id") == 1)
    assert df2.count() == 1
