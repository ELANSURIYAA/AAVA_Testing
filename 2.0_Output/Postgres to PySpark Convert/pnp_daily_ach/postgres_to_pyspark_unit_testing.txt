def test_TC01_happy_path(spark):
    # Happy path: All input DataFrames have valid data
    data = [
        ("2023-06-01", "T1", "John Smith", 5000.0, "CHK1", "CREDIT", "PPD", "desc", "disc", "1841010148", "CompanyA", "SDA1", "RT1", "reason", "DA1"),
    ]
    schema = T.StructType([
        T.StructField("business_dt", T.StringType()),
        T.StructField("transaction_par_nbr", T.StringType()),
        T.StructField("dest_customer_nm", T.StringType()),
        T.StructField("dest_transaction_amt", T.DoubleType()),
        T.StructField("dest_check_nbr", T.StringType()),
        T.StructField("credit_debit_cde", T.StringType()),
        T.StructField("nacha_sec_code", T.StringType()),
        T.StructField("transaction_desc", T.StringType()),
        T.StructField("discretionary_txt", T.StringType()),
        T.StructField("originator_company_id", T.StringType()),
        T.StructField("originator_company_name", T.StringType()),
        T.StructField("src_dest_arrangement_nbr", T.StringType()),
        T.StructField("originating_rt_nbr", T.StringType()),
        T.StructField("desti_return_reason_desc", T.StringType()),
        T.StructField("dest_arrangement_id", T.StringType()),
    ])
    staging_cust360_ach_batch_item = spark.createDataFrame(data, schema)
    tbl_transaction_detail_ach = spark.createDataFrame([], schema)
    tbl_evn_na_merchant_details = spark.createDataFrame([], T.StructType([
        T.StructField("prcs_sys_id_txt", T.StringType())
    ]))
    wxlu_pnp_ach_daily_tmp_s7 = update_tbl_trans_details_ach_0()

    # Step 0: Get max business_dt (simulate as empty)
    dt_max = None

    # Step 1
    wxlu_pnp_ach_daily_tmp_s0 = staging_cust360_ach_batch_item

    # Step 2
    wxlu_pnp_ach_daily_tmp_s1 = wxlu_pnp_ach_daily_tmp_s0.select(
        F.lit("ACH").alias("sor_type"),
        F.col("transaction_par_nbr").alias("trans_id"),
        F.lit("zxx__cust_name").alias("col_name"),
        cust_name_mod(F.col("dest_customer_nm")).alias("zxx__cust_name_mod"),
        F.col("dest_transaction_amt").alias("trans_amt"),
        F.col("business_dt").alias("bus_date")
    )

    # Step 3
    wxlu_pnp_ach_daily_tmp_s2 = pnp_cust_name(wxlu_pnp_ach_daily_tmp_s1)

    # Step 4
    wxlu_pnp_ach_daily_tmp_s3 = wxlu_pnp_ach_daily_tmp_s1.join(
        wxlu_pnp_ach_daily_tmp_s2,
        "zxx__cust_name_mod",
        how="left"
    ).select(
        wxlu_pnp_ach_daily_tmp_s1["*"],
        wxlu_pnp_ach_daily_tmp_s2["zxx__cust_name_mod_new"],
        wxlu_pnp_ach_daily_tmp_s2["is_person"],
        wxlu_pnp_ach_daily_tmp_s2["is_non_person"],
        wxlu_pnp_ach_daily_tmp_s2["is_non_profit"],
        wxlu_pnp_ach_daily_tmp_s2["business_only"],
        F.coalesce(wxlu_pnp_ach_daily_tmp_s2["entity_type"], F.lit("UNIDENTIFIABLE")).alias("entity_type")
    ).distinct()

    # Step 5: entity_type update (simplified for test)
    wxlu_pnp_ach_daily_tmp_s3 = wxlu_pnp_ach_daily_tmp_s3.withColumn(
        "entity_type",
        F.when(
            (F.col("entity_type").isin("PERSON", "PERSON-TBR") == False) &
            (((F.col("is_non_person") <= F.col("is_person")) &
              (F.col("is_non_profit") <= F.col("is_person")) &
              (F.col("trans_amt") < 6000)) |
             ((F.col("is_person") == 0) &
              (F.col("is_non_person") == 0) &
              (F.col("is_non_profit") == 0) &
              (F.col("trans_amt") < 1000))),
            "PERSON-TBR"
        ).otherwise(F.col("entity_type"))
    )

    # Step 6: join with s0
    wxlu_pnp_ach_daily_tmp_s4 = wxlu_pnp_ach_daily_tmp_s0.join(
        wxlu_pnp_ach_daily_tmp_s3,
        wxlu_pnp_ach_daily_tmp_s0["transaction_par_nbr"] == wxlu_pnp_ach_daily_tmp_s3["trans_id"],
        how="left"
    ).select(
        wxlu_pnp_ach_daily_tmp_s0["*"],
        wxlu_pnp_ach_daily_tmp_s3["entity_type"].alias("destination_pnp_cde")
    ).distinct()

    # Assert
    result = wxlu_pnp_ach_daily_tmp_s4.collect()
    assert len(result) == 1
    assert result[0]["destination_pnp_cde"] in ("PERSON", "PERSON-TBR")

def test_TC05_person_regex(spark):
    # Name matches PERSON regex
    data = [
        ("2023-06-01", "T1", "JOHN Q SMITH", 500.0, "CHK1", "CREDIT", "PPD", "desc", "disc", "1841010148", "CompanyA", "SDA1", "RT1", "reason", "DA1"),
    ]
    schema = T.StructType([
        T.StructField("business_dt", T.StringType()),
        T.StructField("transaction_par_nbr", T.StringType()),
        T.StructField("dest_customer_nm", T.StringType()),
        T.StructField("dest_transaction_amt", T.DoubleType()),
        T.StructField("dest_check_nbr", T.StringType()),
        T.StructField("credit_debit_cde", T.StringType()),
        T.StructField("nacha_sec_code", T.StringType()),
        T.StructField("transaction_desc", T.StringType()),
        T.StructField("discretionary_txt", T.StringType()),
        T.StructField("originator_company_id", T.StringType()),
        T.StructField("originator_company_name", T.StringType()),
        T.StructField("src_dest_arrangement_nbr", T.StringType()),
        T.StructField("originating_rt_nbr", T.StringType()),
        T.StructField("desti_return_reason_desc", T.StringType()),
        T.StructField("dest_arrangement_id", T.StringType()),
    ])
    staging_cust360_ach_batch_item = spark.createDataFrame(data, schema)
    wxlu_pnp_ach_daily_tmp_s0 = staging_cust360_ach_batch_item
    wxlu_pnp_ach_daily_tmp_s1 = wxlu_pnp_ach_daily_tmp_s0.select(
        F.lit("ACH").alias("sor_type"),
        F.col("transaction_par_nbr").alias("trans_id"),
        F.lit("zxx__cust_name").alias("col_name"),
        cust_name_mod(F.col("dest_customer_nm")).alias("zxx__cust_name_mod"),
        F.col("dest_transaction_amt").alias("trans_amt"),
        F.col("business_dt").alias("bus_date")
    )
    wxlu_pnp_ach_daily_tmp_s2 = pnp_cust_name(wxlu_pnp_ach_daily_tmp_s1)
    wxlu_pnp_ach_daily_tmp_s3 = wxlu_pnp_ach_daily_tmp_s1.join(
        wxlu_pnp_ach_daily_tmp_s2,
        "zxx__cust_name_mod",
        how="left"
    ).select(
        wxlu_pnp_ach_daily_tmp_s1["*"],
        wxlu_pnp_ach_daily_tmp_s2["zxx__cust_name_mod_new"],
        wxlu_pnp_ach_daily_tmp_s2["is_person"],
        wxlu_pnp_ach_daily_tmp_s2["is_non_person"],
        wxlu_pnp_ach_daily_tmp_s2["is_non_profit"],
        wxlu_pnp_ach_daily_tmp_s2["business_only"],
        F.coalesce(wxlu_pnp_ach_daily_tmp_s2["entity_type"], F.lit("UNIDENTIFIABLE")).alias("entity_type")
    ).distinct()
    person_patterns = [
        r"^[A-Z]{3,} [A-Z] [A-Z]{2,}$",
        r"^[A-Z]{3,} [A-Z]{3,} [A-Z]$",
        r"^[A-Z] [A-Z]{3,} [A-Z]{3,}$",
        r"^[A-Z]{3,} [A-Z]{3,} (SR|JR)$",
        r"^[A-Z]{3,} [A-Z] [A-Z]{3,} (I|II|III|IV|V|SR|JR)$"
    ]
    person_regex = "|".join(person_patterns)
    wxlu_pnp_ach_daily_tmp_s3 = wxlu_pnp_ach_daily_tmp_s3.withColumn(
        "entity_type",
        F.when(
            (F.col("entity_type") == "PERSON-TBR") &
            (F.col("zxx__cust_name_mod_new").rlike(person_regex)),
            "PERSON"
        ).otherwise(F.col("entity_type"))
    )
    result = wxlu_pnp_ach_daily_tmp_s3.collect()
    assert result[0]["entity_type"] in ("PERSON", "PERSON-TBR", "PERSON")
