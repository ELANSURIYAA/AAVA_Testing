-- Description:   Executes DQ validation checks on MEMBERSDATA and logs violations into DQ_ISSUE_LOG
-- =============================================

CREATE OR REPLACE PROCEDURE MY_SAMPLE_DB.PUBLIC.SP_RUN_DQ_CHECKS()
RETURNS STRING
LANGUAGE JAVASCRIPT
EXECUTE AS CALLER
AS
$$
/*
  This procedure dynamically executes DQ checks on MEMBERSDATA table as per recommendations,
  and logs any violations into DQ_ISSUE_LOG with a unique BATCH_ID.
*/

var dqRules = [
    { "Column": "FIRSTNAME", "RuleType": "NotNullCheck", "RuleDescription": "Ensure the first name is provided for every member.", "Severity": "High" },
    { "Column": "LASTNAME", "RuleType": "NotNullCheck", "RuleDescription": "Ensure the last name is provided for every member.", "Severity": "High" },
    { "Column": "COMPANY", "RuleType": "StringLengthCheck", "RuleDescription": "Ensure the company name does not exceed 255 characters.", "Severity": "Medium" },
    { "Column": "ADDRESS", "RuleType": "StringLengthCheck", "RuleDescription": "Ensure the address does not exceed 16777216 characters.", "Severity": "Low" },
    { "Column": "CITY", "RuleType": "ValueCheck", "RuleDescription": "Ensure the city is one of the approved locations: Los Angeles, New York, or London.", "Severity": "High" },
    { "Column": "COUNTRY", "RuleType": "NotNullCheck", "RuleDescription": "Ensure the country is provided for every member.", "Severity": "High" },
    { "Column": "STATE", "RuleType": "StringLengthCheck", "RuleDescription": "Ensure the state name does not exceed 100 characters.", "Severity": "Medium" },
    { "Column": "ZIP", "RuleType": "FormatCheck", "RuleDescription": "Ensure the ZIP code matches the expected format (e.g., numeric or alphanumeric).", "Severity": "Medium" },
    { "Column": "PHONENUMBER1", "RuleType": "NotNullCheck", "RuleDescription": "Ensure the primary phone number is provided.", "Severity": "High" },
    { "Column": "PHONENUMBER2", "RuleType": "UniquenessCheck", "RuleDescription": "Ensure PHONENUMBER2 is distinct from PHONENUMBER1 for each member.", "Severity": "High" },
    { "Column": "EMAIL", "RuleType": "FormatCheck", "RuleDescription": "Ensure the email matches a valid email format.", "Severity": "High" },
    { "Column": "WEB", "RuleType": "FormatCheck", "RuleDescription": "Ensure the website URL matches a valid URL format.", "Severity": "Medium" },
    { "Column": "MEMBER_ID", "RuleType": "UniquenessCheck", "RuleDescription": "Ensure MEMBER_ID is unique for each member.", "Severity": "High" },
    { "Column": "MEMBER_ID", "RuleType": "RangeCheck", "RuleDescription": "Ensure MEMBER_ID is within the range of 10000000 to 99999999.", "Severity": "High" },
    { "Column": "PHONENUMBER1, EMAIL, WEB", "RuleType": "NotNullCheck", "RuleDescription": "Ensure at least one valid form of contact information is provided (Phone, Email, or Website).", "Severity": "High" }
];

// Ensure SEQ_BATCH_ID exists
var createSeqSql = "CREATE SEQUENCE IF NOT EXISTS MY_SAMPLE_DB.PUBLIC.SEQ_BATCH_ID;";
try {
    snowflake.createStatement({sqlText: createSeqSql}).execute();
} catch (err) {
    // Sequence might already exist, ignore error
}

// Get new batch id
var batchStmt = snowflake.createStatement({sqlText: "SELECT MY_SAMPLE_DB.PUBLIC.SEQ_BATCH_ID.NEXTVAL as BATCH_ID"});
var batchIdResult = batchStmt.execute();
batchIdResult.next();
var batchId = batchIdResult.getColumnValue('BATCH_ID');

// Helper to escape single quotes for SQL literals
function esc(str) {
    if (str === null || str === undefined) return '';
    return String(str).replace(/'/g, "''");
}

// DQ Checks implementation
for (var i = 0; i < dqRules.length; i++) {
    var rule = dqRules[i];
    var columns = rule.Column.split(',').map(function(x){return x.trim();});
    var ruleType = rule.RuleType;
    var ruleDesc = rule.RuleDescription;
    var severity = rule.Severity;
    var sql = "";
    var colExpr = columns.map(function(c){return '"' + c + '"';}).join(', ');

    // Build the WHERE clause for failed records
    if (ruleType === "NotNullCheck" && columns.length === 1) {
        sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
              "SELECT 'MEMBERSDATA', '" + esc(columns[0]) + "', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
              "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE " + columns[0] + " IS NULL";
    } else if (ruleType === "NotNullCheck" && columns.length > 1) {
        // At least one must be NOT NULL, so fail if all are NULL
        var nullChecks = columns.map(function(c){return c + " IS NULL";}).join(" AND ");
        sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
              "SELECT 'MEMBERSDATA', '" + esc(columns.join(',')) + "', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
              "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE " + nullChecks;
    } else if (ruleType === "StringLengthCheck") {
        // Extract max length from rule description
        var match = ruleDesc.match(/(\d+)(?= characters?)/);
        var maxLen = match ? match[1] : '0';
        sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
              "SELECT 'MEMBERSDATA', '" + esc(columns[0]) + "', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
              "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE LENGTH(" + columns[0] + ") > " + maxLen;
    } else if (ruleType === "ValueCheck") {
        // Approved values from rule description
        if (columns[0] === "CITY") {
            sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
                  "SELECT 'MEMBERSDATA', 'CITY', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
                  "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE CITY IS NOT NULL AND CITY NOT IN ('Los Angeles', 'New York', 'London')";
        }
    } else if (ruleType === "FormatCheck") {
        if (columns[0] === "ZIP") {
            // Accepts 5 digit or 5-4 digit ZIP, or alphanumeric up to 10 chars
            sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
                  "SELECT 'MEMBERSDATA', 'ZIP', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
                  "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE ZIP IS NOT NULL AND NOT (ZIP RLIKE '^[0-9]{5}(-[0-9]{4})?$' OR ZIP RLIKE '^[A-Za-z0-9]{3,10}$')";
        } else if (columns[0] === "EMAIL") {
            // Simple email regex
            sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
                  "SELECT 'MEMBERSDATA', 'EMAIL', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
                  "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE EMAIL IS NOT NULL AND NOT (EMAIL RLIKE '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$')";
        } else if (columns[0] === "WEB") {
            // Simple URL regex
            sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
                  "SELECT 'MEMBERSDATA', 'WEB', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
                  "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE WEB IS NOT NULL AND NOT (WEB RLIKE '^(https?://)?[A-Za-z0-9.-]+\\.[A-Za-z]{2,}(/.*)?$')";
        }
    } else if (ruleType === "UniquenessCheck") {
        if (columns.length === 1 && columns[0] === "MEMBER_ID") {
            // Duplicates on MEMBER_ID
            sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
                  "SELECT 'MEMBERSDATA', 'MEMBER_ID', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
                  "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE MEMBER_ID IN (SELECT MEMBER_ID FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA GROUP BY MEMBER_ID HAVING COUNT(*) > 1)";
        } else if (columns.length === 2 && columns[0] === "PHONENUMBER2" && columns[1] === "PHONENUMBER1") {
            // PHONENUMBER2 must be distinct from PHONENUMBER1
            sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
                  "SELECT 'MEMBERSDATA', 'PHONENUMBER2', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
                  "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE PHONENUMBER2 IS NOT NULL AND PHONENUMBER2 = PHONENUMBER1";
        }
    } else if (ruleType === "RangeCheck" && columns[0] === "MEMBER_ID") {
        // MEMBER_ID range
        sql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) " +
              "SELECT 'MEMBERSDATA', 'MEMBER_ID', '" + esc(ruleDesc) + "', '" + esc(severity) + "', " + batchId + ", MEMBER_ID " +
              "FROM MY_SAMPLE_DB.PUBLIC.MEMBERSDATA WHERE MEMBER_ID IS NOT NULL AND (MEMBER_ID < 10000000 OR MEMBER_ID > 99999999)";
    }

    if (sql) {
        try {
            snowflake.createStatement({sqlText: sql}).execute();
        } catch (err) {
            var errMsg = "Error executing DQ rule: " + esc(ruleDesc) + " | Error: " + esc(err.message);
            var logErrSql = "INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID) VALUES ('MEMBERSDATA', 'SYSTEM', '" + esc(errMsg) + "', 'High', " + batchId + ", NULL)";
            snowflake.createStatement({sqlText: logErrSql}).execute();
        }
    }
}

return "DQ checks completed. Batch ID: " + batchId;

$$;
