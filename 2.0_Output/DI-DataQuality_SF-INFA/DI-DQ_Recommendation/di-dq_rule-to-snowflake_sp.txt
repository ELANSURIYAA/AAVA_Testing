CREATE OR REPLACE PROCEDURE MY_SAMPLE_DB.PUBLIC.SP_RUN_DQ_CHECKS()
RETURNS STRING
LANGUAGE JAVASCRIPT
EXECUTE AS OWNER
AS
$$
/*
  Executes DQ validation checks on MEMBERSDATA and logs violations in DQ_ISSUE_LOG.
  Dynamically constructs and executes validation queries based on DQ rules.
  Inserts failed records into DQ_ISSUE_LOG with BATCH_ID.
*/

function ensure_batch_id_sequence() {
    var stmt = snowflake.createStatement({
        sqlText: `CREATE SEQUENCE IF NOT EXISTS MY_SAMPLE_DB.PUBLIC.SEQ_BATCH_ID`
    });
    stmt.execute();
}

function get_new_batch_id() {
    var stmt = snowflake.createStatement({
        sqlText: `SELECT MY_SAMPLE_DB.PUBLIC.SEQ_BATCH_ID.NEXTVAL`
    });
    var rs = stmt.execute();
    rs.next();
    return rs.getColumnValue(1);
}

function dq_insert(table, column, rule_desc, severity, batch_id, member_id) {
    var stmt = snowflake.createStatement({
        sqlText: `INSERT INTO MY_SAMPLE_DB.PUBLIC.DQ_ISSUE_LOG
            (TABLE_NAME, COLUMN_NAME, RULE_DESCRIPTION, SEVERITY, BATCH_ID, MEMBER_ID)
            VALUES (?, ?, ?, ?, ?, ?)`,
        binds: [table, column, rule_desc, severity, batch_id, member_id]
    });
    stmt.execute();
}

function dq_bulk_insert(sql, table, column, rule_desc, severity, batch_id) {
    // Insert all failed MEMBER_IDs for the rule
    var stmt = snowflake.createStatement({sqlText: sql});
    var rs = stmt.execute();
    while (rs.next()) {
        var member_id = rs.getColumnValue(1);
        dq_insert(table, column, rule_desc, severity, batch_id, member_id);
    }
}

function escape_sql_literal(str) {
    // Escapes single quotes for safe SQL
    return str.replace(/'/g, "''");
}

function run_checks() {
    var table = 'MEMBERSDATA';
    var schema = 'MY_SAMPLE_DB.PUBLIC';
    var batch_id;

    ensure_batch_id_sequence();
    batch_id = get_new_batch_id();

    // DQ Rules (from DI-DQ_Recommendation.txt)
    var dq_rules = [
        {
            "Column": "FIRSTNAME",
            "Rule Type": "NotNullCheck",
            "Rule Description": "Ensure the first name is provided for every member.",
            "Severity": "High"
        },
        {
            "Column": "LASTNAME",
            "Rule Type": "NotNullCheck",
            "Rule Description": "Ensure the last name is provided for every member.",
            "Severity": "High"
        },
        {
            "Column": "COMPANY",
            "Rule Type": "StringLengthCheck",
            "Rule Description": "Ensure the company name does not exceed 255 characters.",
            "Severity": "Medium"
        },
        {
            "Column": "ADDRESS",
            "Rule Type": "StringLengthCheck",
            "Rule Description": "Ensure the address does not exceed 16777216 characters.",
            "Severity": "Low"
        },
        {
            "Column": "CITY",
            "Rule Type": "ValueCheck",
            "Rule Description": "Ensure the city is one of the approved locations: Los Angeles, New York, or London.",
            "Severity": "High"
        },
        {
            "Column": "COUNTRY",
            "Rule Type": "NotNullCheck",
            "Rule Description": "Ensure the country is provided for every member.",
            "Severity": "High"
        },
        {
            "Column": "STATE",
            "Rule Type": "StringLengthCheck",
            "Rule Description": "Ensure the state name does not exceed 100 characters.",
            "Severity": "Medium"
        },
        {
            "Column": "ZIP",
            "Rule Type": "FormatCheck",
            "Rule Description": "Ensure the ZIP code matches the expected format (e.g., numeric or alphanumeric).",
            "Severity": "Medium"
        },
        {
            "Column": "PHONENUMBER1",
            "Rule Type": "NotNullCheck",
            "Rule Description": "Ensure the primary phone number is provided.",
            "Severity": "High"
        },
        {
            "Column": "PHONENUMBER2",
            "Rule Type": "UniquenessCheck",
            "Rule Description": "Ensure PHONENUMBER2 is distinct from PHONENUMBER1 for each member.",
            "Severity": "High"
        },
        {
            "Column": "EMAIL",
            "Rule Type": "FormatCheck",
            "Rule Description": "Ensure the email matches a valid email format.",
            "Severity": "High"
        },
        {
            "Column": "WEB",
            "Rule Type": "FormatCheck",
            "Rule Description": "Ensure the website URL matches a valid URL format.",
            "Severity": "Medium"
        },
        {
            "Column": "MEMBER_ID",
            "Rule Type": "UniquenessCheck",
            "Rule Description": "Ensure MEMBER_ID is unique for each member.",
            "Severity": "High"
        },
        {
            "Column": "MEMBER_ID",
            "Rule Type": "RangeCheck",
            "Rule Description": "Ensure MEMBER_ID is within the range of 10000000 to 99999999.",
            "Severity": "High"
        },
        {
            "Column": "PHONENUMBER1, EMAIL, WEB",
            "Rule Type": "NotNullCheck",
            "Rule Description": "Ensure at least one valid form of contact information is provided (Phone, Email, or Website).",
            "Severity": "High"
        }
    ];

    for (var i = 0; i < dq_rules.length; i++) {
        var rule = dq_rules[i];
        var col = rule["Column"];
        var rule_type = rule["Rule Type"];
        var desc = rule["Rule Description"];
        var severity = rule["Severity"];
        var sql = "";

        // NotNullCheck (single column)
        if (rule_type === "NotNullCheck" && col.indexOf(",") === -1) {
            sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE " + col + " IS NULL";
            dq_bulk_insert(sql, table, col, desc, severity, batch_id);
        }
        // NotNullCheck (multi-column: at least one must be not null)
        else if (rule_type === "NotNullCheck" && col.indexOf(",") !== -1) {
            var cols = col.split(",").map(function(x){return x.trim();});
            var conds = cols.map(function(c){return c + " IS NOT NULL";}).join(" OR ");
            sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE NOT (" + conds + ")";
            dq_bulk_insert(sql, table, col, desc, severity, batch_id);
        }
        // StringLengthCheck
        else if (rule_type === "StringLengthCheck") {
            var maxlen = 0;
            if (col === "COMPANY") maxlen = 255;
            else if (col === "ADDRESS") maxlen = 16777216;
            else if (col === "STATE") maxlen = 100;
            sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE LENGTH(" + col + ") > " + maxlen;
            dq_bulk_insert(sql, table, col, desc, severity, batch_id);
        }
        // ValueCheck
        else if (rule_type === "ValueCheck") {
            // Only for CITY in this case
            sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE " + col + " IS NULL OR " + col + " NOT IN ('Los Angeles', 'New York', 'London')";
            dq_bulk_insert(sql, table, col, desc, severity, batch_id);
        }
        // FormatCheck
        else if (rule_type === "FormatCheck") {
            if (col === "ZIP") {
                // Accepts numeric or alphanumeric ZIPs (US/UK style)
                sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE ZIP IS NOT NULL AND ZIP NOT RLIKE '^[A-Za-z0-9\\- ]+$'";
                dq_bulk_insert(sql, table, col, desc, severity, batch_id);
            } else if (col === "EMAIL") {
                // Basic email regex
                sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE EMAIL IS NOT NULL AND EMAIL NOT RLIKE '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$'";
                dq_bulk_insert(sql, table, col, desc, severity, batch_id);
            } else if (col === "WEB") {
                // Basic URL regex
                sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE WEB IS NOT NULL AND WEB NOT RLIKE '^(https?://)?[A-Za-z0-9.-]+\\.[A-Za-z]{2,}.*$'";
                dq_bulk_insert(sql, table, col, desc, severity, batch_id);
            }
        }
        // UniquenessCheck
        else if (rule_type === "UniquenessCheck") {
            if (col === "MEMBER_ID") {
                // Duplicate MEMBER_IDs
                sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " GROUP BY MEMBER_ID HAVING COUNT(*) > 1";
                dq_bulk_insert(sql, table, col, desc, severity, batch_id);
            } else if (col === "PHONENUMBER2") {
                // PHONENUMBER2 must be distinct from PHONENUMBER1 for each member
                sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE PHONENUMBER2 = PHONENUMBER1 AND PHONENUMBER2 IS NOT NULL";
                dq_bulk_insert(sql, table, col, desc, severity, batch_id);
            }
        }
        // RangeCheck
        else if (rule_type === "RangeCheck") {
            if (col === "MEMBER_ID") {
                sql = "SELECT MEMBER_ID FROM " + schema + "." + table + " WHERE MEMBER_ID < 10000000 OR MEMBER_ID > 99999999";
                dq_bulk_insert(sql, table, col, desc, severity, batch_id);
            }
        }
    }
    return "DQ checks executed. Batch ID: " + batch_id;
}

try {
    return run_checks();
} catch (err) {
    return "Error running DQ checks: " + err; 
}
$$;
