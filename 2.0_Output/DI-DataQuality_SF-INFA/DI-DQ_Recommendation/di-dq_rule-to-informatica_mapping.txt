<POWERMART>
  <REPOSITORY NAME="MY_SAMPLE_DB" VERSION="180" CODEPAGE="UTF-8">
    <FOLDER NAME="DQ_Mapping" GROUP="" OWNER="DataEngineer" SHARED="NOTSHARED" DESCRIPTION="DQ mapping for MEMBERSDATA table" VERSION="1">
      <!-- Source Definition -->
      <SOURCE NAME="MEMBERSDATA" DATABASETYPE="Flat File" DESCRIPTION="Source table for members data">
        <FIELD NAME="FIRSTNAME" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="LASTNAME" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="COMPANY" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="ADDRESS" DATATYPE="string" LENGTH="16777216"/>
        <FIELD NAME="CITY" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="COUNTRY" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="STATE" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="ZIP" DATATYPE="string" LENGTH="20"/>
        <FIELD NAME="PHONENUMBER1" DATATYPE="string" LENGTH="50"/>
        <FIELD NAME="PHONENUMBER2" DATATYPE="string" LENGTH="50"/>
        <FIELD NAME="EMAIL" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="WEB" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="MEMBER_ID" DATATYPE="integer" PRECISION="8" SCALE="0"/>
      </SOURCE>

      <!-- Expression Transformation for DQ Checks -->
      <TRANSFORMATION NAME="DQ_Expression" TYPE="Expression" DESCRIPTION="DQ rule checks">
        <FIELD NAME="FIRSTNAME" EXPRESSION="FIRSTNAME"/>
        <FIELD NAME="LASTNAME" EXPRESSION="LASTNAME"/>
        <FIELD NAME="COMPANY" EXPRESSION="COMPANY"/>
        <FIELD NAME="ADDRESS" EXPRESSION="ADDRESS"/>
        <FIELD NAME="CITY" EXPRESSION="CITY"/>
        <FIELD NAME="COUNTRY" EXPRESSION="COUNTRY"/>
        <FIELD NAME="STATE" EXPRESSION="STATE"/>
        <FIELD NAME="ZIP" EXPRESSION="ZIP"/>
        <FIELD NAME="PHONENUMBER1" EXPRESSION="PHONENUMBER1"/>
        <FIELD NAME="PHONENUMBER2" EXPRESSION="PHONENUMBER2"/>
        <FIELD NAME="EMAIL" EXPRESSION="EMAIL"/>
        <FIELD NAME="WEB" EXPRESSION="WEB"/>
        <FIELD NAME="MEMBER_ID" EXPRESSION="MEMBER_ID"/>

        <!-- DQ Rule Outputs -->
        <FIELD NAME="DQ_FIRSTNAME_NOTNULL" DATATYPE="integer" EXPRESSION="IIF(ISNULL(FIRSTNAME), 0, 1)"/>
        <FIELD NAME="DQ_LASTNAME_NOTNULL" DATATYPE="integer" EXPRESSION="IIF(ISNULL(LASTNAME), 0, 1)"/>
        <FIELD NAME="DQ_COMPANY_LENGTH" DATATYPE="integer" EXPRESSION="IIF(LENGTH(COMPANY) <= 255, 1, 0)"/>
        <FIELD NAME="DQ_ADDRESS_LENGTH" DATATYPE="integer" EXPRESSION="IIF(LENGTH(ADDRESS) <= 16777216, 1, 0)"/>
        <FIELD NAME="DQ_CITY_APPROVED" DATATYPE="integer" EXPRESSION="IIF(CITY IN ('Los Angeles','New York','London'), 1, 0)"/>
        <FIELD NAME="DQ_COUNTRY_NOTNULL" DATATYPE="integer" EXPRESSION="IIF(ISNULL(COUNTRY), 0, 1)"/>
        <FIELD NAME="DQ_STATE_LENGTH" DATATYPE="integer" EXPRESSION="IIF(LENGTH(STATE) <= 100, 1, 0)"/>
        <FIELD NAME="DQ_ZIP_FORMAT" DATATYPE="integer" EXPRESSION="IIF(REG_MATCH(ZIP, '^[A-Za-z0-9]{1,20}$'), 1, 0)"/>
        <FIELD NAME="DQ_PHONENUMBER1_NOTNULL" DATATYPE="integer" EXPRESSION="IIF(ISNULL(PHONENUMBER1), 0, 1)"/>
        <FIELD NAME="DQ_PHONENUMBER2_UNIQUE" DATATYPE="integer" EXPRESSION="IIF(PHONENUMBER1 != PHONENUMBER2 OR ISNULL(PHONENUMBER2), 1, 0)"/>
        <FIELD NAME="DQ_EMAIL_FORMAT" DATATYPE="integer" EXPRESSION="IIF(REG_MATCH(EMAIL, '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'), 1, 0)"/>
        <FIELD NAME="DQ_WEB_FORMAT" DATATYPE="integer" EXPRESSION="IIF(REG_MATCH(WEB, '^https?://[A-Za-z0-9.-]+\.[A-Za-z]{2,}.*'), 1, 0)"/>
        <FIELD NAME="DQ_MEMBER_ID_UNIQUE" DATATYPE="integer" EXPRESSION="1"/>
        <FIELD NAME="DQ_MEMBER_ID_RANGE" DATATYPE="integer" EXPRESSION="IIF(MEMBER_ID >= 10000000 AND MEMBER_ID <= 99999999, 1, 0)"/>
        <FIELD NAME="DQ_CONTACT_PROVIDED" DATATYPE="integer" EXPRESSION="IIF(NOT ISNULL(PHONENUMBER1) OR NOT ISNULL(EMAIL) OR NOT ISNULL(WEB), 1, 0)"/>
        <!-- DQ Aggregate Score -->
        <FIELD NAME="DQ_ALL_PASS" DATATYPE="integer" EXPRESSION="DQ_FIRSTNAME_NOTNULL AND DQ_LASTNAME_NOTNULL AND DQ_COMPANY_LENGTH AND DQ_ADDRESS_LENGTH AND DQ_CITY_APPROVED AND DQ_COUNTRY_NOTNULL AND DQ_STATE_LENGTH AND DQ_ZIP_FORMAT AND DQ_PHONENUMBER1_NOTNULL AND DQ_PHONENUMBER2_UNIQUE AND DQ_EMAIL_FORMAT AND DQ_WEB_FORMAT AND DQ_MEMBER_ID_RANGE AND DQ_CONTACT_PROVIDED"/>
      </TRANSFORMATION>

      <!-- Aggregator Transformation for Uniqueness Checks -->
      <TRANSFORMATION NAME="DQ_Aggregator" TYPE="Aggregator" DESCRIPTION="Uniqueness checks for MEMBER_ID">
        <FIELD NAME="MEMBER_ID" EXPRESSION="MEMBER_ID"/>
        <FIELD NAME="COUNT_MEMBER_ID" EXPRESSION="COUNT(MEMBER_ID)" GROUPBY="MEMBER_ID"/>
        <FIELD NAME="DQ_MEMBER_ID_UNIQUE" DATATYPE="integer" EXPRESSION="IIF(COUNT_MEMBER_ID = 1, 1, 0)"/>
      </TRANSFORMATION>

      <!-- Router Transformation for Valid/Rejected Rows -->
      <TRANSFORMATION NAME="DQ_Router" TYPE="Router" DESCRIPTION="Route valid and rejected rows">
        <GROUP NAME="VALID_ROWS" CONDITION="DQ_ALL_PASS = 1 AND DQ_MEMBER_ID_UNIQUE = 1"/>
        <GROUP NAME="REJECTED_ROWS" CONDITION="DQ_ALL_PASS = 0 OR DQ_MEMBER_ID_UNIQUE = 0"/>
      </TRANSFORMATION>

      <!-- Target Definitions -->
      <TARGET NAME="MEMBERSDATA_VALID" DATABASETYPE="Flat File" DESCRIPTION="Target for valid members data">

        <FIELD NAME="COMPANY" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="ADDRESS" DATATYPE="string" LENGTH="16777216"/>
        <FIELD NAME="CITY" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="COUNTRY" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="STATE" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="ZIP" DATATYPE="string" LENGTH="20"/>
        <FIELD NAME="PHONENUMBER1" DATATYPE="string" LENGTH="50"/>
        <FIELD NAME="PHONENUMBER2" DATATYPE="string" LENGTH="50"/>
        <FIELD NAME="EMAIL" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="WEB" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="MEMBER_ID" DATATYPE="integer" PRECISION="8" SCALE="0"/>
      </TARGET>
      <TARGET NAME="MEMBERSDATA_REJECTED" DATABASETYPE="Flat File" DESCRIPTION="Target for rejected members data">
        <FIELD NAME="FIRSTNAME" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="LASTNAME" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="COMPANY" DATATYPE="string" LENGTH="255"/>
        <FIELD NAME="ADDRESS" DATATYPE="string" LENGTH="16777216"/>
        <FIELD NAME="CITY" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="COUNTRY" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="STATE" DATATYPE="string" LENGTH="100"/>
        <FIELD NAME="ZIP" DATATYPE="string" LENGTH="20"/>
        <FIELD NAME="PHONENUMBER1" DATATYPE="string" LENGTH="50"/>
        <FIELD NAME="PHONENUMBER2" DATATYPE="string" LENGTH="50"/>
        <FIELD NAME="EMAIL" DATATYPE="string" LENGTH="255"/>

        <!-- Optionally add DQ flags for rejected rows -->
        <FIELD NAME="DQ_FIRSTNAME_NOTNULL" DATATYPE="integer"/>

        <FIELD NAME="DQ_EMAIL_FORMAT" DATATYPE="integer"/>

      </TARGET>

      <!-- Mapping Definition -->
      <MAPPING NAME="DQ_MembersData_Mapping" DESCRIPTION="Mapping for DQ validation of MEMBERSDATA">
        <TRANSFORMATIONS>
          <TRANSFORMATION_REF NAME="DQ_Expression"/>
          <TRANSFORMATION_REF NAME="DQ_Aggregator"/>
          <TRANSFORMATION_REF NAME="DQ_Router"/>
        </TRANSFORMATIONS>
        <CONNECTORS>
          <CONNECTOR FROM="MEMBERSDATA" TO="DQ_Expression"/>
          <CONNECTOR FROM="DQ_Expression" TO="DQ_Aggregator"/>
          <CONNECTOR FROM="DQ_Aggregator" TO="DQ_Router"/>
          <CONNECTOR FROM="DQ_Router.VALID_ROWS" TO="MEMBERSDATA_VALID"/>
          <CONNECTOR FROM="DQ_Router.REJECTED_ROWS" TO="MEMBERSDATA_REJECTED"/>
        </CONNECTORS>
      </MAPPING>
