CleanDLNService.java
Implements the transformation logic from ECL cleanDLN function. */ package com.example.cleandln.service;
import com.example.cleandln.constants.Constants; import com.example.cleandln.patterns.Patterns; import org.slf4j.Logger; import org.slf4j.LoggerFactory; import org.springframework.stereotype.Service;

import java.util.Arrays; import java.util.regex.Matcher; import java.util.regex.Pattern;

@Service public class CleanDLNService { private static final Logger logger = LoggerFactory.getLogger(CleanDLNService.class);

/**
 * Cleans the DLN string based on source and state, mimicking ECL logic.
 * @param dln   Driver License Number (string)
 * @param source Source system (string)
 * @param state  State code (string)
 * @return Cleaned DLN or empty string if not applicable
 */
public String cleanDLN(String dln, String source, String state) {
    if (dln == null) dln = "";
    dln = dln.trim();

    // Remove trailing State Postal Code if it exists
    Pattern trailingSpcPattern = Pattern.compile(Patterns.TRAILING_SPC);
    Matcher matcher = trailingSpcPattern.matcher(dln);
    String trailingSpc = matcher.find() ? matcher.group() : "";

    int trailingSpcCnt = trailingSpc.isEmpty() ? 0 : countOccurrences(dln, trailingSpc);
    int trailingSpcPos = trailingSpc.isEmpty() ? 0 : dln.lastIndexOf(trailingSpc);

    // If DLN has trailing State Postal Code, get substring up to the position before trailing State Postal Code starts
    String cleanDLN1 = trailingSpc.isEmpty() ? dln : dln.substring(0, trailingSpcPos);

    // If source is in the set and state matches, filter out ' ' and '-' from DLN
    String cleanedDLN;
    if (Arrays.asList(Constants.SET_DATA_SOURCE_WITH_DLN).contains(source)) {
        if (Constants.SPC.equals(state)) {
            cleanedDLN = filterOut(cleanDLN1, " -");
        } else {
            cleanedDLN = cleanDLN1;
        }
    } else {
        cleanedDLN = "";
    }
    logger.info("Cleaned DLN for input [dln={}, source={}, state={}] is [{}]", dln, source, state, cleanedDLN);
    return cleanedDLN;
}

// Helper to count occurrences of substring in string
private int countOccurrences(String str, String sub) {
    int count = 0, idx = 0;
    while ((idx = str.indexOf(sub, idx)) != -1) {
        count++;
        idx += sub.length();
    }
    return count;
}

// Helper to filter out characters from a string
private String filterOut(String input, String charsToRemove) {
    StringBuilder sb = new StringBuilder();
    for (char c : input.toCharArray()) {
        if (charsToRemove.indexOf(c) == -1) {
            sb.append(c);
        }
    }
    return sb.toString();
}
}

/*

CleanDLNController.java
REST Controller exposing the cleanDLN endpoint. */ package com.example.cleandln.controller;
import com.example.cleandln.model.CleanDLNRequest; import com.example.cleandln.model.CleanDLNResponse; import com.example.cleandln.service.CleanDLNService; import org.slf4j.Logger; import org.slf4j.LoggerFactory; import org.springframework.http.ResponseEntity; import org.springframework.web.bind.annotation.*;

@RestController @RequestMapping("/api/v1/cleandln") public class CleanDLNController { private static final Logger logger = LoggerFactory.getLogger(CleanDLNController.class);

private final CleanDLNService cleanDLNService;

public CleanDLNController(CleanDLNService cleanDLNService) {
    this.cleanDLNService = cleanDLNService;
}

/**
 * POST endpoint to clean DLN based on source and state.
 * @param request CleanDLNRequest containing dln, source, and state
 * @return CleanDLNResponse with cleaned DLN string
 */
@PostMapping
public ResponseEntity<CleanDLNResponse> cleanDLN(@RequestBody CleanDLNRequest request) {
    logger.info("Received cleanDLN request: {}", request);
    String cleaned = cleanDLNService.cleanDLN(request.getDln(), request.getSource(), request.getState());
    return ResponseEntity.ok(new CleanDLNResponse(cleaned));
}
}
