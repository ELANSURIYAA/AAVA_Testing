```java
package com.example.constants;

import org.junit.jupiter.api.Test;
import java.lang.reflect.Constructor;
import java.lang.reflect.Modifier;

import static org.junit.jupiter.api.Assertions.*;

/**
 * Unit tests for the Constants class.
 */
public class ConstantsTest {

    // TC_001: Verify NOT_FOUND_STATUS constant value
    @Test
    void testNotFoundStatusValue() {
        assertEquals("Not found", Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should be 'Not found'");
    }

    // TC_002: Verify MAX_JOIN_LIMIT constant value
    @Test
    void testMaxJoinLimitValue() {
        assertEquals(2000, Constants.MAX_JOIN_LIMIT, "MAX_JOIN_LIMIT should be 2000");
    }

    // TC_003: Verify NOT_FOUND_STATUS is not null or empty
    @Test
    void testNotFoundStatusNotNullOrEmpty() {
        assertNotNull(Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should not be null");
        assertFalse(Constants.NOT_FOUND_STATUS.isEmpty(), "NOT_FOUND_STATUS should not be empty");
    }

    // TC_004: Verify MAX_JOIN_LIMIT is positive
    @Test
    void testMaxJoinLimitIsPositive() {
        assertTrue(Constants.MAX_JOIN_LIMIT > 0, "MAX_JOIN_LIMIT should be positive");
    }

    // TC_005: Verify Constants class cannot be instantiated
    @Test
    void testConstantsClassCannotBeInstantiated() {
        Constructor<?>[] constructors = Constants.class.getDeclaredConstructors();
        for (Constructor<?> constructor : constructors) {
            assertTrue(Modifier.isPrivate(constructor.getModifiers()),
                "Constructor should be private to prevent instantiation");
        }
        // Attempt to instantiate via reflection (should fail if private)
        try {
            Constructor<Constants> constructor = Constants.class.getDeclaredConstructor();
            constructor.setAccessible(true);
            constructor.newInstance();
            fail("Instantiation should not be allowed for Constants class");
        } catch (Exception e) {
            // Expected: InstantiationException, IllegalAccessException, etc.
            assertTrue(e instanceof IllegalAccessException
                    || e.getCause() instanceof IllegalAccessException
                    || e.getCause() instanceof UnsupportedOperationException,
                "Instantiation should throw IllegalAccessException or similar");
        }
    }

    // TC_006: Verify Constants fields are final and cannot be modified (immutability)
    @Test
    void testConstantsFieldsAreFinal() throws NoSuchFieldException {
        assertTrue(Modifier.isFinal(Constants.class.getField("NOT_FOUND_STATUS").getModifiers()),
            "NOT_FOUND_STATUS should be final");
        assertTrue(Modifier.isFinal(Constants.class.getField("MAX_JOIN_LIMIT").getModifiers()),
            "MAX_JOIN_LIMIT should be final");
    }
}
```
