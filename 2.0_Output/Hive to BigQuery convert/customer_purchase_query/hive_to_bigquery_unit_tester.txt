import pytest
from google.cloud import bigquery

# Helper functions for setup/teardown
def create_table(client, dataset_id, table_id, schema, rows):
    table_ref = f"{client.project}.{dataset_id}.{table_id}"
    client.query(f"DROP TABLE IF EXISTS `{table_ref}`").result()
    table = bigquery.Table(table_ref, schema=schema)
    client.create_table(table)
    if rows:
        errors = client.insert_rows_json(table_ref, rows)
        assert errors == [], f"Errors inserting rows: {errors}"

def drop_table(client, dataset_id, table_id):
    table_ref = f"{client.project}.{dataset_id}.{table_id}"
    client.query(f"DROP TABLE IF EXISTS `{table_ref}`").result()

def run_analysis_query(bq_client, dataset_id):

    with open("customer_purchase_query.txt") as f:

                      .replace("customer_purchase_analysis_report", f"{dataset_id}.customer_purchase_analysis_report")
    bq_client.query(sql).result()

def fetch_report(bq_client, dataset_id):
    query = f"SELECT * FROM `{bq_client.project}.{dataset_id}.customer_purchase_analysis_report`"
