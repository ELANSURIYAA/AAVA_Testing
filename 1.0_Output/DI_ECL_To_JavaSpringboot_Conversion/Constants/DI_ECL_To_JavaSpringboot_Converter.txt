```java
package com.ascendion.ava.constants;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

/**
 * Constants class holding application-wide constant values.
 * Converted from ECL module "Constants".
 */
public class Constants {
    // NotFoundStatus: Equivalent to ECL export NotFoundStatus
    public static final String NOT_FOUND_STATUS = "Not found";
    // MaxJoinLimit: Equivalent to ECL export MaxJoinLimit
    public static final int MAX_JOIN_LIMIT = 2000;
}

/**
 * REST Controller exposing constants for client access.
 */
@RestController
public class ConstantsController {

    /**
     * Returns the NotFoundStatus constant.
     * @return NotFoundStatus string
     */
    @GetMapping("/constants/not-found-status")
    public String getNotFoundStatus() {
        return Constants.NOT_FOUND_STATUS;
    }

    /**
     * Returns the MaxJoinLimit constant.
     * @return MaxJoinLimit integer
     */
    @GetMapping("/constants/max-join-limit")
    public int getMaxJoinLimit() {
        return Constants.MAX_JOIN_LIMIT;
    }
}
```
