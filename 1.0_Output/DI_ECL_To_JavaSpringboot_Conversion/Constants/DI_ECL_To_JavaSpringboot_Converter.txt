package com.ascendion.ava.constants;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * Constants class mapping ECL exported values to Java static fields.
 * Provides REST endpoint to retrieve constant values.
 */
public class Constants {
    // Equivalent to ECL: export NotFoundStatus := 'Not found';
    public static final String NOT_FOUND_STATUS = "Not found";

    // Equivalent to ECL: export MaxJoinLimit := 2000;
    public static final int MAX_JOIN_LIMIT = 2000;
}

@RestController
class ConstantsController {
    private static final Logger logger = LoggerFactory.getLogger(ConstantsController.class);

    /**
     * REST endpoint to retrieve all constant values.
     * @return ConstantsResponse object containing all constants.
     */
    @GetMapping("/api/constants")
    public ConstantsResponse getConstants() {
        logger.info("Fetching constants");
        return new ConstantsResponse(Constants.NOT_FOUND_STATUS, Constants.MAX_JOIN_LIMIT);
    }
}

/**
 * POJO for REST response containing constant values.
 */
class ConstantsResponse {
    private String notFoundStatus;
    private int maxJoinLimit;

    public ConstantsResponse(String notFoundStatus, int maxJoinLimit) {
        this.notFoundStatus = notFoundStatus;
        this.maxJoinLimit = maxJoinLimit;
    }

    public String getNotFoundStatus() {
        return notFoundStatus;
    }

    public int getMaxJoinLimit() {
        return maxJoinLimit;
    }
}

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

/**
 * Spring Boot entry point.
 */
@SpringBootApplication
public class ConstantsApplication {
    public static void main(String[] args) {
        SpringApplication.run(ConstantsApplication.class, args);
    }
}
