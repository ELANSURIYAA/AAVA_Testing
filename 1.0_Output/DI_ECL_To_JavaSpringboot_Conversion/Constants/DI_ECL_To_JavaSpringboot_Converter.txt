 * Application-wide constants converted from ECL module.
 * NotFoundStatus: String constant for "not found" status.
 * MaxJoinLimit: Integer constant for maximum join limit.

public final class Constants {

    // ECL: export NotFoundStatus := 'Not found';

    // ECL: export MaxJoinLimit := 2000;
    public static final int MAX_JOIN_LIMIT = 2000;

    // Private constructor to prevent instantiation

}
