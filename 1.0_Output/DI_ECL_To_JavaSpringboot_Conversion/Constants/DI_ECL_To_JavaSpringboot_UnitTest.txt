Description:   JUnit test cases for Java Spring Boot application-wide constants equivalent to ECL module "Constants" (NotFoundStatus and MaxJoinLimit). Ensures correct constant values and immutability.
=============================================

1. Test Case List:

| Test Case ID | Test Case Description                                              | Expected Outcome                                    |
|--------------|-------------------------------------------------------------------|-----------------------------------------------------|
| TC01         | Verify NOT_FOUND_STATUS constant value                            | Value is "Not found"                                |
| TC02         | Verify MAX_JOIN_LIMIT constant value                              | Value is 2000                                       |
| TC03         | Verify Constants class cannot be instantiated                     | Instantiation throws exception or is not possible    |
| TC04         | Verify NOT_FOUND_STATUS is immutable                             | Value remains "Not found" after attempted change     |
| TC05         | Verify MAX_JOIN_LIMIT is immutable                               | Value remains 2000 after attempted change            |

2. JUnit Test Script

```java
package com.example.constants;

import org.junit.jupiter.api.Test;
import java.lang.reflect.Constructor;
import java.lang.reflect.Modifier;

import static org.junit.jupiter.api.Assertions.*;

class ConstantsTest {

    // TC01: Verify NOT_FOUND_STATUS constant value
    @Test
    void testNotFoundStatusValue() {
        assertEquals("Not found", Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should be 'Not found'");
    }

    // TC02: Verify MAX_JOIN_LIMIT constant value
    @Test
    void testMaxJoinLimitValue() {
        assertEquals(2000, Constants.MAX_JOIN_LIMIT, "MAX_JOIN_LIMIT should be 2000");
    }

    // TC03: Verify Constants class cannot be instantiated
    @Test
    void testConstantsClassCannotBeInstantiated() {
        Constructor<?>[] constructors = Constants.class.getDeclaredConstructors();
        for (Constructor<?> constructor : constructors) {
            assertTrue(Modifier.isPrivate(constructor.getModifiers()), "Constructor should be private");
            constructor.setAccessible(true);
            assertThrows(Exception.class, constructor::newInstance, "Instantiation should throw exception");
        }
    }

    // TC04: Verify NOT_FOUND_STATUS is immutable
    @Test
    void testNotFoundStatusIsImmutable() {
        // Strings are immutable in Java, so reference should not change
        assertEquals("Not found", Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should remain 'Not found'");
    }

    // TC05: Verify MAX_JOIN_LIMIT is immutable
    @Test
    void testMaxJoinLimitIsImmutable() {
        // Primitives are immutable, so value should not change
        assertEquals(2000, Constants.MAX_JOIN_LIMIT, "MAX_JOIN_LIMIT should remain 2000");
    }
}
```
