import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;
import java.lang.reflect.Constructor;
import java.lang.reflect.InvocationTargetException;
import com.ascendion.ava.constants.Constants;

public class ConstantsTest {

    // TC01: Verify NOT_FOUND_STATUS constant value
    @Test
    void testNotFoundStatusValue() {
        assertEquals("Not found", Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should be 'Not found'");
    }

    // TC02: Verify MAX_JOIN_LIMIT constant value
    @Test
    void testMaxJoinLimitValue() {
        assertEquals(2000, Constants.MAX_JOIN_LIMIT, "MAX_JOIN_LIMIT should be 2000");
    }

    // TC03: Ensure Constants class cannot be instantiated
    @Test
    void testConstantsClassCannotBeInstantiated() {
        Constructor<?>[] constructors = Constants.class.getDeclaredConstructors();
        constructors[0].setAccessible(true);
        Exception exception = assertThrows(InvocationTargetException.class, () -> {
            constructors[0].newInstance();
        });
        assertTrue(exception.getCause() instanceof UnsupportedOperationException
                || exception.getCause() == null, "Instantiation should not be allowed");
    }

    // TC04: Verify NOT_FOUND_STATUS is not null or empty
    @Test
    void testNotFoundStatusNotNullOrEmpty() {
        assertNotNull(Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should not be null");
        assertFalse(Constants.NOT_FOUND_STATUS.trim().isEmpty(), "NOT_FOUND_STATUS should not be empty");
    }

    // TC05: Verify MAX_JOIN_LIMIT is a positive integer
    @Test
    void testMaxJoinLimitPositive() {
        assertTrue(Constants.MAX_JOIN_LIMIT > 0, "MAX_JOIN_LIMIT should be positive");
    }
}
