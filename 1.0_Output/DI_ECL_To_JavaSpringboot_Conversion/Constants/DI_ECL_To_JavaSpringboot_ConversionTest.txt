2. **JUnit Test Script for each test case**

```java
package com.ascendion.ava.constants;

import org.junit.jupiter.api.Test;
import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.test.web.servlet.MockMvc;
import static org.junit.jupiter.api.Assertions.*;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;

@WebMvcTest(ConstantsController.class)
class ConstantsControllerTest {

    @Autowired
    private MockMvc mockMvc;

    // TC01: Validate Constants.NOT_FOUND_STATUS value
    @Test
    void testNotFoundStatusConstantValue() {
        assertEquals("Not found", Constants.NOT_FOUND_STATUS, "NOT_FOUND_STATUS should be 'Not found'");
    }

    // TC02: Validate Constants.MAX_JOIN_LIMIT value
    @Test
    void testMaxJoinLimitConstantValue() {
        assertEquals(2000, Constants.MAX_JOIN_LIMIT, "MAX_JOIN_LIMIT should be 2000");
    }

    // TC03: GET /constants/not-found-status endpoint returns correct constant
    @Test
    void testGetNotFoundStatusEndpoint() throws Exception {
        mockMvc.perform(get("/constants/not-found-status"))
                .andExpect(status().isOk())
                .andExpect(content().string("Not found"))
                .andExpect(content().contentTypeCompatibleWith("text/plain"));
    }

    // TC04: GET /constants/max-join-limit endpoint returns correct constant
    @Test
    void testGetMaxJoinLimitEndpoint() throws Exception {
        mockMvc.perform(get("/constants/max-join-limit"))
                .andExpect(status().isOk())
                .andExpect(content().string("2000"))
                .andExpect(content().contentTypeCompatibleWith("text/plain"));
    }

    // TC05: GET /constants/not-found-status endpoint handles missing mapping gracefully
    @Test
    void testNotFoundStatusEndpointNotFound() throws Exception {
        mockMvc.perform(get("/constants/not-found-status-invalid"))
                .andExpect(status().isNotFound());
    }

    // TC06: GET /constants/max-join-limit endpoint handles missing mapping gracefully
    @Test
    void testMaxJoinLimitEndpointNotFound() throws Exception {
        mockMvc.perform(get("/constants/max-join-limit-invalid"))
                .andExpect(status().isNotFound());
    }

    // TC07: Validate constants are immutable
    @Test
    void testConstantsImmutability() {
        // Java final fields can't be reassigned, but let's attempt via reflection
        assertThrows(IllegalAccessException.class, () -> {
            java.lang.reflect.Field field = Constants.class.getField("NOT_FOUND_STATUS");
            field.setAccessible(true);
            field.set(null, "Changed");
        });
    }

    // TC08: Validate controller returns correct content type
    @Test
    void testControllerContentType() throws Exception {
        mockMvc.perform(get("/constants/not-found-status"))
                .andExpect(content().contentTypeCompatibleWith("text/plain"));
        mockMvc.perform(get("/constants/max-join-limit"))
                .andExpect(content().contentTypeCompatibleWith("text/plain"));
    }
}
```
