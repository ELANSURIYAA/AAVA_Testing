-- Session management and control flow commands from Teradata (.LOGON, .IF, .LABEL, .EXIT, .LOGOFF) are omitted, as Snowflake handles connections and errors differently.

CREATE TABLE employee_bkup (
    EmployeeNo INTEGER,
    FirstName CHAR(30),
    LastName CHAR(30),
    DepartmentNo SMALLINT,
    NetPay INTEGER
);

-- Drop table if exists (optional, uncomment if desired)
-- DROP TABLE IF EXISTS employee_bkup;

-- Insert employee data from Employee and Salary tables
INSERT INTO employee_bkup
SELECT
    a.EmployeeNo,
    a.FirstName,
    a.LastName,
    a.DepartmentNo,
    b.NetPay
FROM
    Employee a
    INNER JOIN Salary b
        ON a.EmployeeNo = b.EmployeeNo;
