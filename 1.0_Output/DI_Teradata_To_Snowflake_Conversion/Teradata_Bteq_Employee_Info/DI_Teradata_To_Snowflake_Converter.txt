DROP TABLE IF EXISTS employee_bkup;

CREATE TABLE employee_bkup (
    EmployeeNo INTEGER,
    FirstName VARCHAR(30),
    LastName VARCHAR(30),
    DepartmentNo SMALLINT,
    NetPay INTEGER
);

INSERT INTO employee_bkup
SELECT
    a.EmployeeNo,
    a.FirstName,
    a.LastName,
    a.DepartmentNo,
    b.NetPay
FROM
    Employee a
    INNER JOIN Salary b
        ON a.EmployeeNo = b.EmployeeNo;

-- Note: Teradata session management (.LOGON, .LOGOFF), error handling (.IF ERRORCODE), and procedural commands (.LABEL, .GOTO) are omitted as they are not applicable in Snowflake SQL. Only DDL and DML statements are included.
