import pytest
from pymongo import MongoClient

@pytest.fixture(scope="module")
def mongo_client():
    client = MongoClient("mongodb://localhost:27017/")
    yield client
    client.close()

@pytest.fixture(scope="module")
def setup_database(mongo_client):
    db = mongo_client["test_db"]
    db.Documents.drop()
    db.Sample.drop()
    db.create_collection("Documents", validator={
        "$jsonSchema": {
            "bsonType": "object",
            "required": ["content"],
            "properties": {
                "content": {
                    "bsonType": "string",
                    "description": "Content of the document"
                }
            }
        }
    })
    db.create_collection("Sample", validator={
        "$jsonSchema": {
            "bsonType": "object",
            "required": ["title", "body"],
            "properties": {
                "title": {
                    "bsonType": "string",
                    "description": "Title of the sample document"
                },
                "body": {
                    "bsonType": "string",
                    "description": "Body of the sample document"
                }
            }
        }
    })
    yield db
    db.Documents.drop()
    db.Sample.drop()

def test_insert_documents(setup_database):
    db = setup_database
    result = db.Documents.insert_one({"content": "This is a new document"})
    assert result.inserted_id is not None

def test_empty_document_insertion(setup_database):
    db = setup_database
    with pytest.raises(Exception):
        db.Documents.insert_one({})

def test_missing_fields_insertion(setup_database):
    db = setup_database
    with pytest.raises(Exception):
        db.Documents.insert_one({"title": "Missing content field"})

def test_text_search(setup_database):
    db = setup_database
    db.Documents.insert_many([
        {"content": "Document 1"},
        {"content": "Document 2"}
    ])
    results = db.Documents.find({"$text": {"$search": "Document 1"}})
    assert len(list(results)) == 1

def test_large_data_set_insertion(setup_database):
    db = setup_database
    large_data = [{"content": f"Document {i}"} for i in range(1000)]
    result = db.Documents.insert_many(large_data)
    assert len(result.inserted_ids) == 1000

def test_invalid_document_structure(setup_database):
    db = setup_database
    with pytest.raises(Exception):
        db.Documents.insert_one({"invalid_field": "Invalid content"})

def test_unsupported_query(setup_database):
    db = setup_database
    with pytest.raises(Exception):
        db.Documents.find({"unsupported_field": "value"})

def test_update_document(setup_database):
    db = setup_database
    db.Documents.insert_one({"content": "Old content"})
    db.Documents.update_one({"content": "Old content"}, {"$set": {"content": "Updated content"}})
    updated_doc = db.Documents.find_one({"content": "Updated content"})
    assert updated_doc is not None

def test_delete_document(setup_database):
    db = setup_database
    db.Documents.insert_one({"content": "To be deleted"})
    db.Documents.delete_one({"content": "To be deleted"})
    deleted_doc = db.Documents.find_one({"content": "To be deleted"})
    assert deleted_doc is None

def test_index_and_aggregation(setup_database):
    db = setup_database
    db.Documents.create_index([("content", "text")])
    db.Documents.insert_many([
        {"content": "Aggregation test 1"},
        {"content": "Aggregation test 2"}
    ])
    aggregation_result = db.Documents.aggregate([{"$match": {"content": {"$regex": "Aggregation"}}}])
    assert len(list(aggregation_result)) == 2
