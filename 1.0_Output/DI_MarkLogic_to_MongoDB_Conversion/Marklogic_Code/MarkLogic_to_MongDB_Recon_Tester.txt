# Python Script for Validating Data Migration from MarkLogic to MongoDB

```python
import json
import xml.etree.ElementTree as ET
from pymongo import MongoClient
import requests

# Step 1: Parse MarkLogic XQuery or Structured Document Input
def parse_marklogic_data(xml_data):
    """
    Parse XML data from MarkLogic into a structured format.
    """
    root = ET.fromstring(xml_data)
    parsed_data = []
    for doc in root.findall(".//{http://example.com/ns}doc"):
        content = doc.find("{http://example.com/ns}content").text
        parsed_data.append({"content": content})
    return parsed_data

# Step 2: Parse MongoDB Query Code
def parse_mongodb_data(mongo_data):
    """
    Parse MongoDB data into a structured format.
    """
    return json.loads(mongo_data)

# Step 3: Establish Connections
def connect_to_marklogic(base_url, username, password):
    """
    Establish a connection to MarkLogic using REST API.
    """
    session = requests.Session()
    session.auth = (username, password)
    session.headers.update({"Content-Type": "application/xml"})
    return session

def connect_to_mongodb(uri):
    """
    Establish a connection to MongoDB.
    """
    client = MongoClient(uri)
    return client

# Step 4: Fetch Data from MarkLogic
def fetch_marklogic_data(session, endpoint):
    """
    Fetch data from MarkLogic using REST API.
    """
    response = session.get(endpoint)
    if response.status_code == 200:
        return response.text
    else:
        raise Exception(f"Failed to fetch data from MarkLogic: {response.status_code}")

# Step 5: Transform Data
def transform_data(marklogic_data):
    """
    Transform MarkLogic data to match MongoDB schema.
    """
    transformed_data = []
    for item in marklogic_data:
        transformed_data.append({"content": item["content"]})
    return transformed_data

# Step 6: Load Data into MongoDB
def load_data_to_mongodb(client, database, collection, data):
    """
    Load transformed data into MongoDB.
    """
    db = client[database]
    col = db[collection]
    result = col.insert_many(data)
    return result.inserted_ids

# Step 7: Execute MongoDB Queries
def execute_mongodb_query(client, database, collection, query):
    """
    Execute a query on MongoDB.
    """
    db = client[database]
    col = db[collection]
    results = col.find(query)
    return list(results)

# Step 8: Compare Data
def compare_data(marklogic_data, mongodb_data):
    """
    Compare data between MarkLogic and MongoDB.
    """
    discrepancies = []
    for ml_doc, mongo_doc in zip(marklogic_data, mongodb_data):
        if ml_doc != mongo_doc:
            discrepancies.append({"marklogic": ml_doc, "mongodb": mongo_doc})
    return discrepancies

# Step 9: Generate Report
def generate_report(discrepancies):
    """
    Generate a detailed validation report.
    """
    report = {
        "match_status": "MATCH" if not discrepancies else "NO MATCH",
        "discrepancies": discrepancies,
        "summary": f"{len(discrepancies)} discrepancies found."
    }
    return report

# Main Execution
if __name__ == "__main__":
    # Configuration
    marklogic_base_url = "http://marklogic-server:8000"
    marklogic_username = "admin"
    marklogic_password = "password"
    marklogic_endpoint = "/v1/documents"

    mongodb_uri = "mongodb://localhost:27017/"
    mongodb_database = "test_db"
    mongodb_collection = "Documents"

    # Connect to MarkLogic and MongoDB
    marklogic_session = connect_to_marklogic(marklogic_base_url, marklogic_username, marklogic_password)
    mongodb_client = connect_to_mongodb(mongodb_uri)

    # Fetch and Parse Data
    marklogic_raw_data = fetch_marklogic_data(marklogic_session, marklogic_endpoint)
    marklogic_data = parse_marklogic_data(marklogic_raw_data)

    # Transform and Load Data
    transformed_data = transform_data(marklogic_data)
    load_data_to_mongodb(mongodb_client, mongodb_database, mongodb_collection, transformed_data)

    # Execute Queries and Compare Data
    mongodb_data = execute_mongodb_query(mongodb_client, mongodb_database, mongodb_collection, {})
    discrepancies = compare_data(marklogic_data, mongodb_data)

    # Generate and Print Report
    report = generate_report(discrepancies)
    print(json.dumps(report, indent=4))
```

This script performs all the required steps for validating the data migration from MarkLogic to MongoDB, including parsing, transformation, loading, comparison, and reporting. It ensures data consistency, integrity, and completeness across both databases.
