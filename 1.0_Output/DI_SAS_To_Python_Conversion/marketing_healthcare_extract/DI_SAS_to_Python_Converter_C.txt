import pandas as pd
import numpy as np
import logging
from sqlalchemy import create_engine

# Configure logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

# Step 1: Secure Teradata connection using metadata-authenticated libref
logging.info("Step 1: Establishing connection to Teradata database.")
try:
    engine = create_engine('teradatasql://username:password@your_server_name/your_tdpid')
    healthcare_plans = pd.read_sql("SELECT * FROM tdata.healthcare_plans", engine)
    provider_network = pd.read_sql("SELECT * FROM tdata.provider_network", engine)
    logging.info("Connection established and data extracted successfully.")
except Exception as e:
    logging.error(f"Error connecting to database: {e}")
    raise

# Step 2: Create work table with enriched data and business rules
logging.info("Step 2: Creating work table with enriched data and business rules.")
try:
    marketing_extract = healthcare_plans.merge(provider_network, on='plan_id', how='left')
    marketing_extract['member_full_name'] = marketing_extract['member_first_name'].str.strip() + ' ' + marketing_extract['member_last_name'].str.strip()
    marketing_extract['age_group'] = np.where(
        (marketing_extract['member_age'] >= 25) & (marketing_extract['member_age'] <= 35), 'YOUNG_ADULT',
        np.where(
            (marketing_extract['member_age'] >= 36) & (marketing_extract['member_age'] <= 50), 'MID_AGE',
            np.where(marketing_extract['member_age'] > 50, 'SENIOR', 'UNKNOWN')
        )
    )
    marketing_extract['marketing_segment'] = marketing_extract['plan_type'].str.upper() + '-' + marketing_extract['region']
    marketing_extract['is_campaign_eligible'] = np.where(
        (marketing_extract['plan_status'] == 'ACTIVE') &
        (marketing_extract['member_age'] >= 25) & (marketing_extract['member_age'] <= 60) &
        (marketing_extract['plan_type'].isin(['PPO', 'HMO'])) &
        (marketing_extract['region'].isin(['WEST', 'SOUTH'])) &
        (marketing_extract['network_type'] == 'IN_NETWORK'),
        1, 0
    )
    logging.info("Work table created successfully.")
except Exception as e:
    logging.error(f"Error creating work table: {e}")
    raise

# Step 3: Filter only eligible members
logging.info("Step 3: Filtering only eligible members.")
try:
    final_campaign_list = marketing_extract[marketing_extract['is_campaign_eligible'] == 1]
    logging.info("Eligible members filtered successfully.")
except Exception as e:
    logging.error(f"Error filtering eligible members: {e}")
    raise

# Step 4: Export final file
logging.info("Step 4: Exporting final file.")
try:
    outpath = '/folders/myfolders/final_marketing_campaign_extract_2025.csv'
    final_campaign_list.to_csv(outpath, index=False)
    logging.info(f"Final file exported successfully to {outpath}.")
except Exception as e:
    logging.error(f"Error exporting final file: {e}")
    raise
