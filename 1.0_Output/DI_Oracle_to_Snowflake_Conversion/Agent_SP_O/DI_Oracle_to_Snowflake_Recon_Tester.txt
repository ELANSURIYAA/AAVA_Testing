```python
import os
import cx_Oracle
import pandas as pd
import pyarrow.parquet as pq
from snowflake.connector import connect
from datetime import datetime
import logging

# Setup logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

# Environment variables for credentials
ORACLE_USER = os.getenv('ORACLE_USER')
ORACLE_PASSWORD = os.getenv('ORACLE_PASSWORD')
ORACLE_DSN = os.getenv('ORACLE_DSN')
SNOWFLAKE_USER = os.getenv('SNOWFLAKE_USER')
SNOWFLAKE_PASSWORD = os.getenv('SNOWFLAKE_PASSWORD')
SNOWFLAKE_ACCOUNT = os.getenv('SNOWFLAKE_ACCOUNT')
SNOWFLAKE_WAREHOUSE = os.getenv('SNOWFLAKE_WAREHOUSE')
SNOWFLAKE_DATABASE = os.getenv('SNOWFLAKE_DATABASE')
SNOWFLAKE_SCHEMA = os.getenv('SNOWFLAKE_SCHEMA')
AZURE_ADLS_PATH = os.getenv('AZURE_ADLS_PATH')

# Oracle connection
def connect_to_oracle():
    try:
        connection = cx_Oracle.connect(user=ORACLE_USER, password=ORACLE_PASSWORD, dsn=ORACLE_DSN)
        logging.info("Connected to Oracle database.")
        return connection
    except Exception as e:
        logging.error(f"Error connecting to Oracle: {e}")
        raise

# Snowflake connection
def connect_to_snowflake():
    try:
        connection = connect(
            user=SNOWFLAKE_USER,
            password=SNOWFLAKE_PASSWORD,
            account=SNOWFLAKE_ACCOUNT,
            warehouse=SNOWFLAKE_WAREHOUSE,
            database=SNOWFLAKE_DATABASE,
            schema=SNOWFLAKE_SCHEMA
        )
        logging.info("Connected to Snowflake.")
        return connection
    except Exception as e:
        logging.error(f"Error connecting to Snowflake: {e}")
        raise

# Execute Oracle stored procedure
def execute_oracle_procedure(connection, procedure_name):
    try:
        cursor = connection.cursor()
        cursor.callproc(procedure_name)
        logging.info(f"Executed Oracle procedure: {procedure_name}")
    except Exception as e:
        logging.error(f"Error executing Oracle procedure: {e}")
        raise

# Export Oracle table to Parquet
def export_to_parquet(connection, table_name):
    try:
        query = f"SELECT * FROM {table_name}"
        df = pd.read_sql(query, con=connection)
        timestamp = datetime.now().strftime('%Y%m%d%H%M%S')
        parquet_file = f"{table_name}_{timestamp}.parquet"
        df.to_parquet(parquet_file, engine='pyarrow')
        logging.info(f"Exported table {table_name} to Parquet: {parquet_file}")
        return parquet_file
    except Exception as e:
        logging.error(f"Error exporting table {table_name} to Parquet: {e}")
        raise

# Transfer Parquet files to Azure ADLS
def transfer_to_adls(parquet_file):
    try:
        # Assuming Azure CLI is configured for ADLS
        os.system(f"az storage blob upload --file {parquet_file} --container-name mycontainer --name {parquet_file}")
        logging.info(f"Transferred {parquet_file} to Azure ADLS.")
    except Exception as e:
        logging.error(f"Error transferring {parquet_file} to Azure ADLS: {e}")
        raise

# Create Snowflake external table
def create_external_table(connection, table_name, parquet_file):
    try:
        query = f"""
        CREATE OR REPLACE EXTERNAL TABLE {table_name}_ext
        WITH LOCATION = '{AZURE_ADLS_PATH}/{parquet_file}'
        FILE_FORMAT = (TYPE = 'PARQUET')
        AUTO_REFRESH = TRUE;
        """
        cursor = connection.cursor()
        cursor.execute(query)
        logging.info(f"Created external table {table_name}_ext in Snowflake.")
    except Exception as e:
        logging.error(f"Error creating external table {table_name}_ext: {e}")
        raise

# Compare data between Oracle and Snowflake
def compare_data(oracle_connection, snowflake_connection, table_name):
    try:
        oracle_query = f"SELECT * FROM {table_name}"
        snowflake_query = f"SELECT * FROM {table_name}_ext"

        oracle_df = pd.read_sql(oracle_query, con=oracle_connection)
        snowflake_df = pd.read_sql(snowflake_query, con=snowflake_connection)

        if oracle_df.equals(snowflake_df):
            logging.info(f"Data match for table {table_name}.")
            return "MATCH"
        else:
            logging.warning(f"Data mismatch for table {table_name}.")
            return "NO MATCH"
    except Exception as e:
        logging.error(f"Error comparing data for table {table_name}: {e}")
        raise

# Main function
def main():
    try:
        # Step 1: Connect to Oracle and Snowflake
        oracle_connection = connect_to_oracle()
        snowflake_connection = connect_to_snowflake()

        # Step 2: Execute Oracle procedure
        execute_oracle_procedure(oracle_connection, 'LOAD_GOLD_AGENTS')

        # Step 3: Export Oracle table to Parquet
        parquet_file = export_to_parquet(oracle_connection, 'GOLD_AGENTS_D')

        # Step 4: Transfer Parquet to ADLS
        transfer_to_adls(parquet_file)

        # Step 5: Create Snowflake external table
        create_external_table(snowflake_connection, 'GOLD_AGENTS_D', parquet_file)

        # Step 6: Compare data
        match_status = compare_data(oracle_connection, snowflake_connection, 'GOLD_AGENTS_D')
        logging.info(f"Comparison result: {match_status}")

    except Exception as e:
        logging.error(f"Error in migration process: {e}")
    finally:
        oracle_connection.close()
        snowflake_connection.close()

if __name__ == "__main__":
    main()
```
