import pytest
import pandas as pd
from pyspark.sql import SparkSession
from pyspark.sql.types import StructType, StructField, StringType, DecimalType
from pyspark.sql.utils import AnalysisException

# Sample schema based on ABAP structure
schema = StructType([
    StructField("bukrs", StringType(), True),
    StructField("fiscyear", StringType(), True),
    StructField("costcenter", StringType(), True),
    StructField("gl_account", StringType(), True),
    StructField("amount", StringType(), True),
    StructField("currency", StringType(), True),
    StructField("posting_date", StringType(), True)
])

@pytest.fixture(scope="module")
def spark():
    return SparkSession.builder.master("local[1]").appName("ABAPtoPySparkTest").getOrCreate()

def test_TC01_file_reading(spark):
    # Test reading a sample CSV file
    try:
        df = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
        assert df.count() > 0
    except Exception as e:
        pytest.fail(f"File reading failed: {e}")

def test_TC02_data_transformation(spark):
    # Test splitting and mapping logic
    df = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
    first_row = df.first()
    assert len(first_row) == 7
    assert all([first_row[i] is not None for i in range(7)])

def test_TC03_error_handling_file_not_found(spark):
    # Test error handling for missing file
    with pytest.raises(AnalysisException):
        spark.read.csv("tests/data
onexistent.csv", schema=schema, header=False).count()

def test_TC04_append_records(spark):
    # Test appending logic (union)
    df1 = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
    df2 = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
    df_combined = df1.union(df2)
    assert df_combined.count() == 2 * df1.count()

def test_TC05_data_insertion(spark, tmp_path):
    # Test DataFrame write
    df = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
    output_path = str(tmp_path / "output")
    df.write.mode("overwrite").parquet(output_path)
    df2 = spark.read.parquet(output_path)
    assert df2.count() == df.count()

def test_TC06_transaction_handling(spark, tmp_path):
    # Simulate commit/rollback by exception
    df = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
    output_path = str(tmp_path / "output2")
    try:
        df.write.mode("overwrite").parquet(output_path)
        committed = True
    except Exception:
        committed = False
    assert committed

def test_TC07_type_and_null_handling(spark):
    # Test type conversion and NULL handling
    df = spark.read.csv("tests/data/finance_data_with_nulls.csv", schema=schema, header=False)
    for row in df.collect():
        assert isinstance(row['bukrs'], (str, type(None)))
        assert isinstance(row['amount'], (str, type(None)))

def test_TC08_performance_optimizations(spark):
    # Test partitioning/caching
    df = spark.read.csv("tests/data/finance_data.csv", schema=schema, header=False)
    df = df.repartition(2)
    df.cache()
    assert df.is_cached

def test_TC09_legacy_constructs_removed():
    # Ensure no ABAP constructs remain in PySpark code
    with open("converted_pyspark_script.py") as f:
        code = f.read()
    assert "sy-subrc" not in code
    assert "WRITE" not in code
    assert "APPEND" not in code
